33 static void add_bytes_altivec(uint8_t *dst, uint8_t *src, ptrdiff_t w)
34 {
35  register int i;
36  register vector unsigned char vdst, vsrc;
37 
38  /* dst and src are 16 bytes-aligned (guaranteed). */
39  for (i = 0; i + 15 < w; i += 16) {
40  vdst = vec_ld(i, (unsigned char *) dst);
41  vsrc = vec_ld(i, (unsigned char *) src);
42  vdst = vec_add(vsrc, vdst);
43  vec_st(vdst, i, (unsigned char *) dst);
44  }
45  /* If w is not a multiple of 16. */
46  for (; i < w; i++)
47  dst[i] = src[i];
48 }
