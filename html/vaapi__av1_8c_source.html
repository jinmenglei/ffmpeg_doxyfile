34 static int8_t vaapi_av1_get_bit_depth_idx(AVCodecContext *avctx)
35 {
36  AV1DecContext *s = avctx->priv_data;
37  const AV1RawSequenceHeader *seq = s->raw_seq;
38  int8_t bit_depth = 8;
39 
40  if (seq->seq_profile == 2 && seq->color_config.high_bitdepth)
41  bit_depth = seq->color_config.twelve_bit ? 12 : 10;
42  else if (seq->seq_profile <= 2)
43  bit_depth = seq->color_config.high_bitdepth ? 10 : 8;
44  else {
45  av_log(avctx, AV_LOG_ERROR,
46  "Couldn't get bit depth from profile:%d.\n", seq->seq_profile);
47  return -1;
48  }
49  return bit_depth == 8 ? 0 : bit_depth == 10 ? 1 : 2;
50 }

195  }
196  for (int i = 0; i < frame_header->tile_rows; i++) {

225 static int vaapi_av1_decode_slice(AVCodecContext *avctx,
226  const uint8_t *buffer,
227  uint32_t size)
228 {
229  const AV1DecContext *s = avctx->priv_data;
230  VAAPIDecodePicture *pic = s->cur_frame.hwaccel_picture_private;
231  VASliceParameterBufferAV1 slice_param;
232  int err = 0;
233 
234  for (int i = s->tg_start; i <= s->tg_end; i++) {
235  memset(&slice_param, 0, sizeof(VASliceParameterBufferAV1));
236 
237  slice_param = (VASliceParameterBufferAV1) {
238  .slice_data_size = s->tile_group_info[i].tile_size,
239  .slice_data_offset = s->tile_group_info[i].tile_offset,
240  .slice_data_flag = VA_SLICE_DATA_FLAG_ALL,
241  .tile_row = s->tile_group_info[i].tile_row,
242  .tile_column = s->tile_group_info[i].tile_column,
243  .tg_start = s->tg_start,
244  .tg_end = s->tg_end,
245  };
246 
247  err = ff_vaapi_decode_make_slice_buffer(avctx, pic, &slice_param,
248  sizeof(VASliceParameterBufferAV1),
249  buffer,
250  s->tile_group_info[i].tile_size);
251  if (err) {
252  ff_vaapi_decode_cancel(avctx, pic);
253  return err;
254  }
255  }
256 
257  return 0;
