66 static int failing_write_packet(AVFormatContext *avf, AVPacket *pkt)
67 {
68  FailingMuxerContext *ctx = avf->priv_data;
69  int ret = 0;
70  if (!pkt) {
71  ctx->flush_count++;
72  } else {
73  FailingMuxerPacketData *data = (FailingMuxerPacketData*) pkt->data;
74 
75  if (!data->recover_after) {
76  data->ret = 0;
77  } else {
78  data->recover_after--;
79  }
80 
81  ret = data->ret;
82 
83  if (data->sleep_time) {
84  int64_t slept = 0;
85  while (slept < data->sleep_time) {
86  if (ff_check_interrupt(&avf->interrupt_callback))
87  return AVERROR_EXIT;
88  av_usleep(SLEEPTIME_10_MS);
89  slept += SLEEPTIME_10_MS;
90  }
91  }
92 
93  if (!ret) {
94  ctx->pts_written[ctx->pts_written_nr++] = pkt->pts;
95  av_packet_unref(pkt);
96  }
97  }
98  return ret;
99 }
