64 do { \
65  char *next, *cur = str, sep; \
66  int ret; \
67  \
68  if (str && strchr(str, ',')) { \
69  av_log(ctx, AV_LOG_WARNING, "This syntax is deprecated, use '|' to "\
70  "separate %s.\n", desc); \
71  sep = ','; \
72  } else \
73  sep = '|'; \
74  \
75  while (cur) { \
76  type fmt; \
77  next = strchr(cur, sep); \
78  if (next) \
79  *next++ = 0; \
80  \

85  if ((ret = add_to_list(&list, fmt)) < 0) { \
86  return ret; \
87  } \

90  } \
