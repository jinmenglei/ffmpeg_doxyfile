34 static int avs2_probe(const AVProbeData *p)
35 {
36  uint32_t code = -1, hds = 0, pic = 0, seq = 0;
37  uint8_t state = 0;
38  const uint8_t *ptr = p->buf, *end = p->buf + p->buf_size, *sqb = 0;
39  if (AV_RB32(p->buf) != 0x1B0){
40  return 0;
41  }
42 
43  while (ptr < end) {
44  ptr = avpriv_find_start_code(ptr, end, &code);
45  state = code & 0xFF;
46  if ((code & 0xffffff00) == 0x100) {
47  if (AVS2_ISUNIT(state)) {
48  if (sqb && !hds) {
49  hds = ptr - sqb;
50  }
51  if (AVS2_ISSQH(state)) {
52  if (!AVS2_ISPROFILE(*ptr))
53  return 0;
54  sqb = ptr;
55  seq++;
56  } else if (AVS2_ISPIC(state)) {
57  pic++;
58  } else if (AVS2_ISEND(state)) {
59  break;
60  }
61  }
62  }
63  }
64  if (seq && hds >= 21 && pic){
65  return AVPROBE_SCORE_EXTENSION + 2; // more than cavs
66  }
67 
68  return 0;
69 }
