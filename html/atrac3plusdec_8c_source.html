81 static av_cold int set_channel_params(ATRAC3PContext *ctx,
82  AVCodecContext *avctx)
83 {
84  memset(ctx->channel_blocks, 0, sizeof(ctx->channel_blocks));
85 
86  switch (avctx->channels) {
87  case 1:
88  if (avctx->channel_layout != AV_CH_FRONT_LEFT)
89  avctx->channel_layout = AV_CH_LAYOUT_MONO;
90 
91  ctx->num_channel_blocks = 1;
92  ctx->channel_blocks[0] = CH_UNIT_MONO;
93  break;
94  case 2:
95  avctx->channel_layout = AV_CH_LAYOUT_STEREO;
96  ctx->num_channel_blocks = 1;
97  ctx->channel_blocks[0] = CH_UNIT_STEREO;
98  break;
99  case 3:
100  avctx->channel_layout = AV_CH_LAYOUT_SURROUND;
101  ctx->num_channel_blocks = 2;
102  ctx->channel_blocks[0] = CH_UNIT_STEREO;
103  ctx->channel_blocks[1] = CH_UNIT_MONO;
104  break;
105  case 4:
106  avctx->channel_layout = AV_CH_LAYOUT_4POINT0;
107  ctx->num_channel_blocks = 3;
108  ctx->channel_blocks[0] = CH_UNIT_STEREO;
109  ctx->channel_blocks[1] = CH_UNIT_MONO;
110  ctx->channel_blocks[2] = CH_UNIT_MONO;
111  break;
112  case 6:
113  avctx->channel_layout = AV_CH_LAYOUT_5POINT1_BACK;
114  ctx->num_channel_blocks = 4;
115  ctx->channel_blocks[0] = CH_UNIT_STEREO;
116  ctx->channel_blocks[1] = CH_UNIT_MONO;
117  ctx->channel_blocks[2] = CH_UNIT_STEREO;
118  ctx->channel_blocks[3] = CH_UNIT_MONO;
119  break;
120  case 7:
121  avctx->channel_layout = AV_CH_LAYOUT_6POINT1_BACK;
122  ctx->num_channel_blocks = 5;
123  ctx->channel_blocks[0] = CH_UNIT_STEREO;
124  ctx->channel_blocks[1] = CH_UNIT_MONO;

147 static av_cold int atrac3p_decode_init(AVCodecContext *avctx)
148 {
149  ATRAC3PContext *ctx = avctx->priv_data;
150  int i, ch, ret;
151 
152  if (!avctx->block_align) {
153  av_log(avctx, AV_LOG_ERROR, "block_align is not set\n");
154  return AVERROR(EINVAL);
155  }
156 
157  ff_atrac3p_init_vlcs();
158 
159  /* initialize IPQF */
160  ff_mdct_init(&ctx->ipqf_dct_ctx, 5, 1, 32.0 / 32768.0);
161 
162  ff_atrac3p_init_imdct(avctx, &ctx->mdct_ctx);
163 
164  ff_atrac_init_gain_compensation(&ctx->gainc_ctx, 6, 2);
165 
166  ff_atrac3p_init_wave_synth();
167 
168  if ((ret = set_channel_params(ctx, avctx)) < 0)
169  return ret;
170 
171  ctx->my_channel_layout = avctx->channel_layout;
172 
173  ctx->ch_units = av_mallocz_array(ctx->num_channel_blocks, sizeof(*ctx->ch_units));
174  ctx->fdsp = avpriv_float_dsp_alloc(avctx->flags & AV_CODEC_FLAG_BITEXACT);
175 
176  if (!ctx->ch_units || !ctx->fdsp) {
177  return AVERROR(ENOMEM);
178  }
179 
180  for (i = 0; i < ctx->num_channel_blocks; i++) {
181  for (ch = 0; ch < 2; ch++) {
182  ctx->ch_units[i].channels[ch].ch_num = ch;
183  ctx->ch_units[i].channels[ch].wnd_shape = &ctx->ch_units[i].channels[ch].wnd_shape_hist[0][0];
184  ctx->ch_units[i].channels[ch].wnd_shape_prev = &ctx->ch_units[i].channels[ch].wnd_shape_hist[1][0];
185  ctx->ch_units[i].channels[ch].gain_data = &ctx->ch_units[i].channels[ch].gain_data_hist[0][0];
186  ctx->ch_units[i].channels[ch].gain_data_prev = &ctx->ch_units[i].channels[ch].gain_data_hist[1][0];
187  ctx->ch_units[i].channels[ch].tones_info = &ctx->ch_units[i].channels[ch].tones_info_hist[0][0];
188  ctx->ch_units[i].channels[ch].tones_info_prev = &ctx->ch_units[i].channels[ch].tones_info_hist[1][0];
189  }
190 
191  ctx->ch_units[i].waves_info = &ctx->ch_units[i].wave_synth_hist[0];
192  ctx->ch_units[i].waves_info_prev = &ctx->ch_units[i].wave_synth_hist[1];
193  }
194 
195  avctx->sample_fmt = AV_SAMPLE_FMT_FLTP;
196 
197  return 0;

200 static void decode_residual_spectrum(ATRAC3PContext *ctx, Atrac3pChanUnitCtx *ch_unit,
201  float out[2][ATRAC3P_FRAME_SAMPLES],
202  int num_channels,
203  AVCodecContext *avctx)
204 {
205  int i, sb, ch, qu, nspeclines, RNG_index;
206  float *dst, q;
207  int16_t *src;
208  /* calculate RNG table index for each subband */
209  int sb_RNG_index[ATRAC3P_SUBBANDS] = { 0 };
210 
211  if (ch_unit->mute_flag) {
212  for (ch = 0; ch < num_channels; ch++)
213  memset(out[ch], 0, ATRAC3P_FRAME_SAMPLES * sizeof(*out[ch]));
214  return;
215  }
216 
217  for (qu = 0, RNG_index = 0; qu < ch_unit->used_quant_units; qu++)
218  RNG_index += ch_unit->channels[0].qu_sf_idx[qu] +
219  ch_unit->channels[1].qu_sf_idx[qu];
220 
221  for (sb = 0; sb < ch_unit->num_coded_subbands; sb++, RNG_index += 128)
222  sb_RNG_index[sb] = RNG_index & 0x3FC;
223 
224  /* inverse quant and power compensation */
225  for (ch = 0; ch < num_channels; ch++) {
226  /* clear channel's residual spectrum */
227  memset(out[ch], 0, ATRAC3P_FRAME_SAMPLES * sizeof(*out[ch]));
228 
229  for (qu = 0; qu < ch_unit->used_quant_units; qu++) {
230  src = &ch_unit->channels[ch].spectrum[ff_atrac3p_qu_to_spec_pos[qu]];
231  dst = &out[ch][ff_atrac3p_qu_to_spec_pos[qu]];
232  nspeclines = ff_atrac3p_qu_to_spec_pos[qu + 1] -
233  ff_atrac3p_qu_to_spec_pos[qu];
234 
235  if (ch_unit->channels[ch].qu_wordlen[qu] > 0) {
236  q = ff_atrac3p_sf_tab[ch_unit->channels[ch].qu_sf_idx[qu]] *
237  ff_atrac3p_mant_tab[ch_unit->channels[ch].qu_wordlen[qu]];
238  for (i = 0; i < nspeclines; i++)
239  dst[i] = src[i] * q;
240  }
241  }
242 
243  for (sb = 0; sb < ch_unit->num_coded_subbands; sb++)
244  ff_atrac3p_power_compensation(ch_unit, ctx->fdsp, ch, &out[ch][0],
245  sb_RNG_index[sb], sb);
246  }
247 
248  if (ch_unit->unit_type == CH_UNIT_STEREO) {
249  for (sb = 0; sb < ch_unit->num_coded_subbands; sb++) {
250  if (ch_unit->swap_channels[sb]) {
251  for (i = 0; i < ATRAC3P_SUBBAND_SAMPLES; i++)
252  FFSWAP(float, out[0][sb * ATRAC3P_SUBBAND_SAMPLES + i],
253  out[1][sb * ATRAC3P_SUBBAND_SAMPLES + i]);
254  }
255 
256  /* flip coefficients' sign if requested */
257  if (ch_unit->negate_coeffs[sb])
258  for (i = 0; i < ATRAC3P_SUBBAND_SAMPLES; i++)
259  out[1][sb * ATRAC3P_SUBBAND_SAMPLES + i] = -(out[1][sb * ATRAC3P_SUBBAND_SAMPLES + i]);
260  }
261  }
262 }

264 static void reconstruct_frame(ATRAC3PContext *ctx, Atrac3pChanUnitCtx *ch_unit,
265  int num_channels, AVCodecContext *avctx)
266 {
267  int ch, sb;
268 
269  for (ch = 0; ch < num_channels; ch++) {
270  for (sb = 0; sb < ch_unit->num_subbands; sb++) {
271  /* inverse transform and windowing */
272  ff_atrac3p_imdct(ctx->fdsp, &ctx->mdct_ctx,
273  &ctx->samples[ch][sb * ATRAC3P_SUBBAND_SAMPLES],
274  &ctx->mdct_buf[ch][sb * ATRAC3P_SUBBAND_SAMPLES],
275  (ch_unit->channels[ch].wnd_shape_prev[sb] << 1) +
276  ch_unit->channels[ch].wnd_shape[sb], sb);
277 
278  /* gain compensation and overlapping */
279  ff_atrac_gain_compensation(&ctx->gainc_ctx,
280  &ctx->mdct_buf[ch][sb * ATRAC3P_SUBBAND_SAMPLES],
281  &ch_unit->prev_buf[ch][sb * ATRAC3P_SUBBAND_SAMPLES],
282  &ch_unit->channels[ch].gain_data_prev[sb],
283  &ch_unit->channels[ch].gain_data[sb],
284  ATRAC3P_SUBBAND_SAMPLES,
285  &ctx->time_buf[ch][sb * ATRAC3P_SUBBAND_SAMPLES]);
286  }
287 
288  /* zero unused subbands in both output and overlapping buffers */
289  memset(&ch_unit->prev_buf[ch][ch_unit->num_subbands * ATRAC3P_SUBBAND_SAMPLES],
290  0,
291  (ATRAC3P_SUBBANDS - ch_unit->num_subbands) *
292  ATRAC3P_SUBBAND_SAMPLES *
293  sizeof(ch_unit->prev_buf[ch][ch_unit->num_subbands * ATRAC3P_SUBBAND_SAMPLES]));
294  memset(&ctx->time_buf[ch][ch_unit->num_subbands * ATRAC3P_SUBBAND_SAMPLES],
295  0,
296  (ATRAC3P_SUBBANDS - ch_unit->num_subbands) *
297  ATRAC3P_SUBBAND_SAMPLES *
298  sizeof(ctx->time_buf[ch][ch_unit->num_subbands * ATRAC3P_SUBBAND_SAMPLES]));
299 
300  /* resynthesize and add tonal signal */
301  if (ch_unit->waves_info->tones_present ||
302  ch_unit->waves_info_prev->tones_present) {
303  for (sb = 0; sb < ch_unit->num_subbands; sb++)
304  if (ch_unit->channels[ch].tones_info[sb].num_wavs ||
305  ch_unit->channels[ch].tones_info_prev[sb].num_wavs) {
306  ff_atrac3p_generate_tones(ch_unit, ctx->fdsp, ch, sb,
307  &ctx->time_buf[ch][sb * 128]);
308  }
309  }
310 
311  /* subband synthesis and acoustic signal output */
312  ff_atrac3p_ipqf(&ctx->ipqf_dct_ctx, &ch_unit->ipqf_ctx[ch],
313  &ctx->time_buf[ch][0], &ctx->outp_buf[ch][0]);
314  }
315 
316  /* swap window shape and gain control buffers. */
317  for (ch = 0; ch < num_channels; ch++) {
318  FFSWAP(uint8_t *, ch_unit->channels[ch].wnd_shape,
319  ch_unit->channels[ch].wnd_shape_prev);
320  FFSWAP(AtracGainInfo *, ch_unit->channels[ch].gain_data,
321  ch_unit->channels[ch].gain_data_prev);
322  FFSWAP(Atrac3pWavesData *, ch_unit->channels[ch].tones_info,
323  ch_unit->channels[ch].tones_info_prev);
324  }
325 
326  FFSWAP(Atrac3pWaveSynthParams *, ch_unit->waves_info, ch_unit->waves_info_prev);
327 }

329 static int atrac3p_decode_frame(AVCodecContext *avctx, void *data,
330  int *got_frame_ptr, AVPacket *avpkt)
331 {
332  ATRAC3PContext *ctx = avctx->priv_data;
333  AVFrame *frame = data;
334  int i, ret, ch_unit_id, ch_block = 0, out_ch_index = 0, channels_to_process;
335  float **samples_p = (float **)frame->extended_data;
336 
337  frame->nb_samples = ATRAC3P_FRAME_SAMPLES;
338  if ((ret = ff_get_buffer(avctx, frame, 0)) < 0)
339  return ret;
340 
341  if ((ret = init_get_bits8(&ctx->gb, avpkt->data, avpkt->size)) < 0)
342  return ret;
343 
344  if (get_bits1(&ctx->gb)) {
345  av_log(avctx, AV_LOG_ERROR, "Invalid start bit!\n");
346  return AVERROR_INVALIDDATA;
347  }
348 
349  while (get_bits_left(&ctx->gb) >= 2 &&
350  (ch_unit_id = get_bits(&ctx->gb, 2)) != CH_UNIT_TERMINATOR) {
351  if (ch_unit_id == CH_UNIT_EXTENSION) {
352  avpriv_report_missing_feature(avctx, "Channel unit extension");
353  return AVERROR_PATCHWELCOME;
354  }
355  if (ch_block >= ctx->num_channel_blocks ||
356  ctx->channel_blocks[ch_block] != ch_unit_id) {
357  av_log(avctx, AV_LOG_ERROR,
358  "Frame data doesn't match channel configuration!\n");
359  return AVERROR_INVALIDDATA;
360  }
361 
362  ctx->ch_units[ch_block].unit_type = ch_unit_id;
363  channels_to_process = ch_unit_id + 1;
364 
365  if ((ret = ff_atrac3p_decode_channel_unit(&ctx->gb,
366  &ctx->ch_units[ch_block],
367  channels_to_process,
368  avctx)) < 0)
369  return ret;
370 
371  decode_residual_spectrum(ctx, &ctx->ch_units[ch_block], ctx->samples,
372  channels_to_process, avctx);
373  reconstruct_frame(ctx, &ctx->ch_units[ch_block],
374  channels_to_process, avctx);
375 
376  for (i = 0; i < channels_to_process; i++)
377  memcpy(samples_p[out_ch_index + i], ctx->outp_buf[i],
378  ATRAC3P_FRAME_SAMPLES * sizeof(**samples_p));
379 
380  ch_block++;
381  out_ch_index += channels_to_process;
382  }
383 
384  *got_frame_ptr = 1;
385 
386  return avctx->codec_id == AV_CODEC_ID_ATRAC3P ? FFMIN(avctx->block_align, avpkt->size) : avpkt->size;
387 }
