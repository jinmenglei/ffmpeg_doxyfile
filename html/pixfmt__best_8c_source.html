42 static enum AVPixelFormat find_best(enum AVPixelFormat pixfmt)
43 {
44  enum AVPixelFormat best = AV_PIX_FMT_NONE;
45  int i;
46  for (i = 0; i < FF_ARRAY_ELEMS(pixfmt_list); i++)
47  best = av_find_best_pix_fmt_of_2(best, pixfmt_list[i],
48  pixfmt, 0, NULL);
49  return best;
50 }
