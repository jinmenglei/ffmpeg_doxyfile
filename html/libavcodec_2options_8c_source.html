107 static int init_context_defaults(AVCodecContext *s, const AVCodec *codec)
108 {
109  int flags=0;
110  memset(s, 0, sizeof(AVCodecContext));
111 
112  s->av_class = &av_codec_context_class;
113 
114  s->codec_type = codec ? codec->type : AVMEDIA_TYPE_UNKNOWN;
115  if (codec) {
116  s->codec = codec;
117  s->codec_id = codec->id;
118  }
119 
120  if(s->codec_type == AVMEDIA_TYPE_AUDIO)
121  flags= AV_OPT_FLAG_AUDIO_PARAM;
122  else if(s->codec_type == AVMEDIA_TYPE_VIDEO)
123  flags= AV_OPT_FLAG_VIDEO_PARAM;
124  else if(s->codec_type == AVMEDIA_TYPE_SUBTITLE)
125  flags= AV_OPT_FLAG_SUBTITLE_PARAM;
126  av_opt_set_defaults2(s, flags, flags);
127 
128  s->time_base = (AVRational){0,1};
129  s->framerate = (AVRational){ 0, 1 };
130  s->pkt_timebase = (AVRational){ 0, 1 };
131  s->get_buffer2 = avcodec_default_get_buffer2;
132  s->get_format = avcodec_default_get_format;
133  s->execute = avcodec_default_execute;
134  s->execute2 = avcodec_default_execute2;
135  s->sample_aspect_ratio = (AVRational){0,1};
136  s->pix_fmt = AV_PIX_FMT_NONE;
137  s->sw_pix_fmt = AV_PIX_FMT_NONE;
138  s->sample_fmt = AV_SAMPLE_FMT_NONE;
139 
140  s->reordered_opaque = AV_NOPTS_VALUE;
141  if(codec && codec->priv_data_size){
142  if(!s->priv_data){
143  s->priv_data= av_mallocz(codec->priv_data_size);
144  if (!s->priv_data) {
145  return AVERROR(ENOMEM);
146  }
147  }
148  if(codec->priv_class){
149  *(const AVClass**)s->priv_data = codec->priv_class;
150  av_opt_set_defaults(s->priv_data);
151  }
152  }
153  if (codec && codec->defaults) {
154  int ret;
155  const AVCodecDefault *d = codec->defaults;
156  while (d->key) {
157  ret = av_opt_set(s, d->key, d->value, 0);
158  av_assert0(ret >= 0);
159  d++;
160  }
161  }
162  return 0;
163 }

231 int avcodec_copy_context(AVCodecContext *dest, const AVCodecContext *src)
232 {
233  const AVCodec *orig_codec = dest->codec;
234  uint8_t *orig_priv_data = dest->priv_data;
235 
236  if (avcodec_is_open(dest)) { // check that the dest context is uninitialized
237  av_log(dest, AV_LOG_ERROR,
238  "Tried to copy AVCodecContext %p into already-initialized %p\n",
239  src, dest);
240  return AVERROR(EINVAL);
241  }
242 
243  copy_context_reset(dest);
244 
245  memcpy(dest, src, sizeof(*dest));
246  av_opt_copy(dest, src);
247 
248  dest->priv_data = orig_priv_data;
249  dest->codec = orig_codec;
250 
251  if (orig_priv_data && src->codec && src->codec->priv_class &&
252  dest->codec && dest->codec->priv_class)
253  av_opt_copy(orig_priv_data, src->priv_data);
254 
255 
256  /* set values specific to opened codecs back to their default state */
257  dest->slice_offset = NULL;
258  dest->hwaccel = NULL;
259  dest->internal = NULL;
260 #if FF_API_CODED_FRAME
261 FF_DISABLE_DEPRECATION_WARNINGS
262  dest->coded_frame = NULL;
263 FF_ENABLE_DEPRECATION_WARNINGS
264 #endif
265 
266  /* reallocate values that should be allocated separately */
267  dest->extradata = NULL;
268  dest->coded_side_data = NULL;
269  dest->intra_matrix = NULL;
270  dest->inter_matrix = NULL;
271  dest->rc_override = NULL;
272  dest->subtitle_header = NULL;
273  dest->hw_frames_ctx = NULL;
274  dest->hw_device_ctx = NULL;
275  dest->nb_coded_side_data = 0;
276 
277 #define alloc_and_copy_or_fail(obj, size, pad) \
278  if (src->obj && size > 0) { \
279  dest->obj = av_malloc(size + pad); \
280  if (!dest->obj) \
281  goto fail; \
282  memcpy(dest->obj, src->obj, size); \
283  if (pad) \
284  memset(((uint8_t *) dest->obj) + size, 0, pad); \
285  }
286  alloc_and_copy_or_fail(extradata, src->extradata_size,
287  AV_INPUT_BUFFER_PADDING_SIZE);
288  dest->extradata_size = src->extradata_size;
289  alloc_and_copy_or_fail(intra_matrix, 64 * sizeof(int16_t), 0);
290  alloc_and_copy_or_fail(inter_matrix, 64 * sizeof(int16_t), 0);
291  alloc_and_copy_or_fail(rc_override, src->rc_override_count * sizeof(*src->rc_override), 0);
292  alloc_and_copy_or_fail(subtitle_header, src->subtitle_header_size, 1);
293  av_assert0(dest->subtitle_header_size == src->subtitle_header_size);
294 #undef alloc_and_copy_or_fail
295 
296  if (src->hw_frames_ctx) {
297  dest->hw_frames_ctx = av_buffer_ref(src->hw_frames_ctx);
298  if (!dest->hw_frames_ctx)
299  goto fail;
300  }
301 
302  return 0;
303 
304 fail:
305  copy_context_reset(dest);
306  return AVERROR(ENOMEM);
307 }
