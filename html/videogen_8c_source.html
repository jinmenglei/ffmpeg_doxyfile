30 static unsigned int myrnd(unsigned int *seed_ptr, int n)
31 {

33 
34  seed = *seed_ptr;
35  seed = (seed * 314159) + 1;
36  if (n == 256) {

52 /* cosine approximate with 1-x^2 */
53 static int int_cos(int a)
54 {
55  int v, neg;
56  a = a & (FRAC_ONE - 1);
57  if (a >= (FRAC_ONE / 2))
58  a = FRAC_ONE - a;
59  neg = 0;
60  if (a > (FRAC_ONE / 4)) {
61  neg = -1;
62  a = (FRAC_ONE / 2) - a;
63  }
64  v = FRAC_ONE - ((a * a) >> 4);
65  v = (v ^ neg) - neg;
66  return v;
67 }
