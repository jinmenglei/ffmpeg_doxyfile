34 static uint32_t read_offs(AVCodecContext *avctx, GetBitContext *gb, uint32_t size, const char *err_msg){
35  uint32_t offs= get_bits_long(gb, 32);
36  if(offs >= size){
37  av_log(avctx, AV_LOG_WARNING, err_msg, offs, size);
38  return 0;
39  }
40  return offs;
41 }

43 static int mjpegb_decode_frame(AVCodecContext *avctx,
44  void *data, int *got_frame,
45  AVPacket *avpkt)
46 {
47  const uint8_t *buf = avpkt->data;
48  int buf_size = avpkt->size;
49  MJpegDecodeContext *s = avctx->priv_data;
50  const uint8_t *buf_end, *buf_ptr;
51  GetBitContext hgb; /* for the header */
52  uint32_t dqt_offs, dht_offs, sof_offs, sos_offs, second_field_offs;
53  uint32_t field_size, sod_offs;
54  int ret;
55 
56  buf_ptr = buf;
57  buf_end = buf + buf_size;
58  s->got_picture = 0;
59  s->adobe_transform = -1;
60 
61 read_header:
62  /* reset on every SOI */
63  s->restart_interval = 0;
64  s->restart_count = 0;
65  s->mjpb_skiptosod = 0;
66 
67  if (buf_end - buf_ptr >= 1 << 28)
68  return AVERROR_INVALIDDATA;
69 
70  init_get_bits(&hgb, buf_ptr, /*buf_size*/(buf_end - buf_ptr)*8);
71 
72  skip_bits(&hgb, 32); /* reserved zeros */
73 
74  if (get_bits_long(&hgb, 32) != MKBETAG('m','j','p','g')) {
75  av_log(avctx, AV_LOG_WARNING, "not mjpeg-b (bad fourcc)\n");
76  return AVERROR_INVALIDDATA;
77  }
78 
79  field_size = get_bits_long(&hgb, 32); /* field size */
80  av_log(avctx, AV_LOG_DEBUG, "field size: 0x%"PRIx32"\n", field_size);
81  skip_bits(&hgb, 32); /* padded field size */
82  second_field_offs = read_offs(avctx, &hgb, buf_end - buf_ptr, "second_field_offs is %d and size is %d\n");
83  av_log(avctx, AV_LOG_DEBUG, "second field offs: 0x%"PRIx32"\n",
84  second_field_offs);
85 
86  dqt_offs = read_offs(avctx, &hgb, buf_end - buf_ptr, "dqt is %d and size is %d\n");
87  av_log(avctx, AV_LOG_DEBUG, "dqt offs: 0x%"PRIx32"\n", dqt_offs);
88  if (dqt_offs) {
89  init_get_bits(&s->gb, buf_ptr+dqt_offs, (buf_end - (buf_ptr+dqt_offs))*8);
90  s->start_code = DQT;
91  ret = ff_mjpeg_decode_dqt(s);
92  if (ret < 0 && (avctx->err_recognition & AV_EF_EXPLODE))
93  return ret;
94  }
95 
96  dht_offs = read_offs(avctx, &hgb, buf_end - buf_ptr, "dht is %d and size is %d\n");
97  av_log(avctx, AV_LOG_DEBUG, "dht offs: 0x%"PRIx32"\n", dht_offs);
98  if (dht_offs) {
99  init_get_bits(&s->gb, buf_ptr+dht_offs, (buf_end - (buf_ptr+dht_offs))*8);
100  s->start_code = DHT;
101  ff_mjpeg_decode_dht(s);
102  }
103 
104  sof_offs = read_offs(avctx, &hgb, buf_end - buf_ptr, "sof is %d and size is %d\n");
105  av_log(avctx, AV_LOG_DEBUG, "sof offs: 0x%"PRIx32"\n", sof_offs);
106  if (sof_offs) {
107  init_get_bits(&s->gb, buf_ptr+sof_offs, (buf_end - (buf_ptr+sof_offs))*8);
108  s->start_code = SOF0;
109  if ((ret = ff_mjpeg_decode_sof(s)) < 0)
110  return ret;
111  }
112 
113  sos_offs = read_offs(avctx, &hgb, buf_end - buf_ptr, "sos is %d and size is %d\n");
114  av_log(avctx, AV_LOG_DEBUG, "sos offs: 0x%"PRIx32"\n", sos_offs);
115  sod_offs = read_offs(avctx, &hgb, buf_end - buf_ptr, "sof is %d and size is %d\n");
116  av_log(avctx, AV_LOG_DEBUG, "sod offs: 0x%"PRIx32"\n", sod_offs);
117  if (sos_offs) {
118  init_get_bits(&s->gb, buf_ptr + sos_offs,
119  8 * FFMIN(field_size, buf_end - buf_ptr - sos_offs));
120  s->mjpb_skiptosod = (sod_offs - sos_offs - show_bits(&s->gb, 16));
121  s->start_code = SOS;
122  ret = ff_mjpeg_decode_sos(s, NULL, 0, NULL);
123  if (ret < 0 && (avctx->err_recognition & AV_EF_EXPLODE))
124  return ret;
125  }
126 
127  if (s->interlaced) {
128  s->bottom_field ^= 1;
129  /* if not bottom field, do not output image yet */
130  if (s->bottom_field != s->interlace_polarity && second_field_offs) {
131  buf_ptr = buf + second_field_offs;
132  goto read_header;
133  }
134  }
135 
136  //XXX FIXME factorize, this looks very similar to the EOI code
137 
138  if(!s->got_picture) {
139  av_log(avctx, AV_LOG_WARNING, "no picture\n");
140  return buf_size;
141  }
142 
143  if ((ret = av_frame_ref(data, s->picture_ptr)) < 0)
144  return ret;
145  *got_frame = 1;
146 
147  if (!s->lossless && avctx->debug & FF_DEBUG_QP) {
148  av_log(avctx, AV_LOG_DEBUG, "QP: %d\n",
149  FFMAX3(s->qscale[0], s->qscale[1], s->qscale[2]));
150  }
151 
152  return buf_size;
153 }
