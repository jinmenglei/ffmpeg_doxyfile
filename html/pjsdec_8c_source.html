36 static int pjs_probe(const AVProbeData *p)
37 {
38  char c;
39  int64_t start, end;
40  const unsigned char *ptr = p->buf;
41 
42  if (sscanf(ptr, "%"SCNd64",%"SCNd64",%c", &start, &end, &c) == 3) {
43  size_t q1pos = strcspn(ptr, "\"");
44  size_t q2pos = q1pos + strcspn(ptr + q1pos + 1, "\"") + 1;
45  if (strcspn(ptr, "\r\n") > q2pos)
46  return AVPROBE_SCORE_MAX;
47  }
48  return 0;
49 }

51 static int64_t read_ts(char **line, int *duration)
52 {
53  int64_t start, end;
54 
55  if (sscanf(*line, "%"SCNd64",%"SCNd64, &start, &end) == 2) {
56  *line += strcspn(*line, "\"");
57  *line += !!**line;
58  if (end < start || end - (uint64_t)start > INT_MAX)
59  return AV_NOPTS_VALUE;
60  *duration = end - start;
61  return start;
62  }
63  return AV_NOPTS_VALUE;
64 }

66 static int pjs_read_header(AVFormatContext *s)
67 {
68  PJSContext *pjs = s->priv_data;
69  AVStream *st = avformat_new_stream(s, NULL);
70 
71  if (!st)
72  return AVERROR(ENOMEM);
73  avpriv_set_pts_info(st, 64, 1, 10);
74  st->codecpar->codec_type = AVMEDIA_TYPE_SUBTITLE;
75  st->codecpar->codec_id = AV_CODEC_ID_PJS;
76 
77  while (!avio_feof(s->pb)) {
78  char line[4096];
79  char *p = line;
80  const int64_t pos = avio_tell(s->pb);
81  int len = ff_get_line(s->pb, line, sizeof(line));
82  int64_t pts_start;
83  int duration;
84 
85  if (!len)
86  break;
87 
88  line[strcspn(line, "\r\n")] = 0;
89 
90  pts_start = read_ts(&p, &duration);
91  if (pts_start != AV_NOPTS_VALUE) {
92  AVPacket *sub;
93 
94  p[strcspn(p, "\"")] = 0;
95  sub = ff_subtitles_queue_insert(&pjs->q, p, strlen(p), 0);
96  if (!sub) {
97  ff_subtitles_queue_clean(&pjs->q);
98  return AVERROR(ENOMEM);
99  }
100  sub->pos = pos;
101  sub->pts = pts_start;
102  sub->duration = duration;
103  }
104  }
105 
106  ff_subtitles_queue_finalize(s, &pjs->q);
107  return 0;
108 }
