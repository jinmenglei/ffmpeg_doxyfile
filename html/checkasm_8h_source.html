144  * This is done by clobbering the stack with junk around the stack pointer and calling the
145  * assembly function through checked_call() with added dummy arguments which forces all
146  * real arguments to be passed on the stack and not in registers. For 32-bit arguments the
147  * upper half of the 64-bit register locations on the stack will now contain junk which will
148  * cause misbehaving functions to either produce incorrect output or segfault. Note that
149  * even though this works extremely well in practice, it's technically not guaranteed

233 /* Benchmark the function */
234 #define bench_new(...)\
235  do {\
236  if (checkasm_bench_func()) {\
237  struct CheckasmPerf *perf = checkasm_get_perf_context();\
238  av_unused const int sysfd = perf->sysfd;\
239  func_type *tfunc = func_new;\
240  uint64_t tsum = 0;\
241  int ti, tcount = 0;\
242  uint64_t t = 0; \
243  for (ti = 0; ti < BENCH_RUNS; ti++) {\
244  PERF_START(t);\
245  tfunc(__VA_ARGS__);\
246  tfunc(__VA_ARGS__);\
247  tfunc(__VA_ARGS__);\
248  tfunc(__VA_ARGS__);\
249  PERF_STOP(t);\
250  if (t*tcount <= tsum*4 && ti > 0) {\
251  tsum += t;\
252  tcount++;\
253  }\
254  }\
255  emms_c();\
256  perf->cycles += t;\
257  perf->iterations++;\
258  }\
259  } while (0)
