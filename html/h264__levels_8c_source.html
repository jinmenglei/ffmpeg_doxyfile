66 // We are only ever interested in the NAL bitrate factor.
67 static int h264_get_br_factor(int profile_idc)
68 {
69  int i;
70  for (i = 0; i < FF_ARRAY_ELEMS(h264_br_factors); i++) {
71  if (h264_br_factors[i].profile_idc == profile_idc)
72  return h264_br_factors[i].cpb_br_nal_factor;
73  }
74  // Default to the non-high profile value if not specified.
75  return 1200;
76 }

90 const H264LevelDescriptor *ff_h264_guess_level(int profile_idc,
91  int64_t bitrate,
92  int framerate,
93  int width, int height,
94  int max_dec_frame_buffering)
95 {
96  int width_mbs = (width + 15) / 16;
97  int height_mbs = (height + 15) / 16;
98  int no_cs3f = !(profile_idc == 66 ||
99  profile_idc == 77 ||
100  profile_idc == 88);
101  int i;
102 
103  for (i = 0; i < FF_ARRAY_ELEMS(h264_levels); i++) {
104  const H264LevelDescriptor *level = &h264_levels[i];
105 
106  if (level->constraint_set3_flag && no_cs3f)
107  continue;
108 
109  if (bitrate > (int64_t)level->max_br * h264_get_br_factor(profile_idc))
110  continue;
111 
112  if (width_mbs * height_mbs > level->max_fs)
113  continue;
114  if (width_mbs * width_mbs > 8 * level->max_fs)
115  continue;
116  if (height_mbs * height_mbs > 8 * level->max_fs)
117  continue;
118 
119  if (width_mbs && height_mbs) {
120  int max_dpb_frames =
121  FFMIN(level->max_dpb_mbs / (width_mbs * height_mbs), 16);
122  if (max_dec_frame_buffering > max_dpb_frames)
123  continue;
124 
125  if (framerate > (level->max_mbps / (width_mbs * height_mbs)))
126  continue;
127  }
128 
129  return level;
130  }
131 
132  // No usable levels found - frame is too big or bitrate is too high.
133  return NULL;
134 }
