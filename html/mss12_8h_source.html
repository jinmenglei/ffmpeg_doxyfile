104 #define ARITH_GET_BIT(prefix) \
105 static int prefix ## _get_bit(ArithCoder *c) \
106 { \
107  int range = c->high - c->low + 1; \
108  int bit = 2 * c->value - c->low >= c->high; \
109  \
110  if (bit) \
111  c->low += range >> 1; \
112  else \
113  c->high = c->low + (range >> 1) - 1; \
114  \
115  prefix ## _normalise(c); \
116  \
117  return bit; \
118 }
