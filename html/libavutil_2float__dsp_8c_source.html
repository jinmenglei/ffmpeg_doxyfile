75 static void vector_fmul_window_c(float *dst, const float *src0,
76  const float *src1, const float *win, int len)
77 {
78  int i, j;
79 
80  dst += len;
81  win += len;
82  src0 += len;
83 
84  for (i = -len, j = len - 1; i < 0; i++, j--) {
85  float s0 = src0[i];
86  float s1 = src1[j];
87  float wi = win[i];
88  float wj = win[j];
89  dst[i] = s0 * wj - s1 * wi;
90  dst[j] = s0 * wi + s1 * wj;
91  }
92 }
