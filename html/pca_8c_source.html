91  for(i=0; i<=j; i++){

93  pca->covariance[j + i*n] -= pca->mean[i] * pca->mean[j];
94  pca->covariance[i + j*n] = pca->covariance[j + i*n];
95  }

104  for(j=i+1; j<n; j++)
105  sum += fabs(pca->covariance[j + i*n]);
106

109  double maxvalue= -1;
110  for(j=i; j<n; j++){
111  if(eigenvalue[j] > maxvalue){
112  maxvalue= eigenvalue[j];
113  k= j;
114  }
115  }

123  }

125  }

164  }
