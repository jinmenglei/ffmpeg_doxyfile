69 // Generate a sine window.
70 av_cold void AAC_RENAME(ff_sine_window_init)(INTFLOAT *window, int n) {
71  int i;
72  for(i = 0; i < n; i++)
73  window[i] = SIN_FIX(sinf((i + 0.5) * (M_PI / (2.0 * n))));
74 }
