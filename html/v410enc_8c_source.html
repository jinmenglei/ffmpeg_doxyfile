41 static int v410_encode_frame(AVCodecContext *avctx, AVPacket *pkt,
42  const AVFrame *pic, int *got_packet)
43 {
44  uint8_t *dst;
45  uint16_t *y, *u, *v;
46  uint32_t val;
47  int i, j, ret;
48 
49  if ((ret = ff_alloc_packet2(avctx, pkt, avctx->width * avctx->height * 4,
50  avctx->width * avctx->height * 4)) < 0)
51  return ret;
52  dst = pkt->data;
53 
54 #if FF_API_CODED_FRAME
55 FF_DISABLE_DEPRECATION_WARNINGS
56  avctx->coded_frame->key_frame = 1;
57  avctx->coded_frame->pict_type = AV_PICTURE_TYPE_I;
58 FF_ENABLE_DEPRECATION_WARNINGS
59 #endif
60 
61  y = (uint16_t *)pic->data[0];
62  u = (uint16_t *)pic->data[1];
63  v = (uint16_t *)pic->data[2];
64 
65  for (i = 0; i < avctx->height; i++) {
66  for (j = 0; j < avctx->width; j++) {
67  val = u[j] << 2;
68  val |= y[j] << 12;
69  val |= (uint32_t) v[j] << 22;
70  AV_WL32(dst, val);
71  dst += 4;
72  }
73  y += pic->linesize[0] >> 1;
74  u += pic->linesize[1] >> 1;
75  v += pic->linesize[2] >> 1;
76  }
77 
78  pkt->flags |= AV_PKT_FLAG_KEY;
79  *got_packet = 1;
80  return 0;
81 }
