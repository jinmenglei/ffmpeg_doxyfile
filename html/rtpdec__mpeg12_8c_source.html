26 static int mpeg_parse_packet(AVFormatContext *ctx, PayloadContext *data,
27  AVStream *st, AVPacket *pkt, uint32_t *timestamp,
28  const uint8_t *buf, int len, uint16_t seq,
29  int flags)
30 {
31  unsigned int h;
32  int ret;
33  if (len <= 4)
34  return AVERROR_INVALIDDATA;
35  h = AV_RB32(buf);
36  buf += 4;
37  len -= 4;
38  if (st->codecpar->codec_type == AVMEDIA_TYPE_VIDEO && h & (1 << 26)) {
39  /* MPEG-2 */
40  if (len <= 4)
41  return AVERROR_INVALIDDATA;
42  buf += 4;
43  len -= 4;
44  }
45  if ((ret = av_new_packet(pkt, len)) < 0)
46  return ret;
47  memcpy(pkt->data, buf, len);
48  pkt->stream_index = st->index;
49  return 0;
50 }
