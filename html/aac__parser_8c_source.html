31  int *need_next_header, int *new_frame_start)
32 {
33  GetBitContext bits;
34  AACADTSHeaderInfo hdr;
35  int size;
36  union {
37  uint64_t u64;
38  uint8_t u8[8 + AV_INPUT_BUFFER_PADDING_SIZE];
39  } tmp;
40 
41  tmp.u64 = av_be2ne64(state);
42  init_get_bits(&bits, tmp.u8 + 8 - AV_AAC_ADTS_HEADER_SIZE,
43  AV_AAC_ADTS_HEADER_SIZE * 8);
44 
45  if ((size = ff_adts_header_parse(&bits, &hdr)) < 0)
46  return 0;
47  *need_next_header = 0;
48  *new_frame_start = 1;

51  hdr_info->samples = hdr.samples;
52  hdr_info->bit_rate = hdr.bit_rate;

54 }
