341 {
342  const AVPixFmtDescriptor *desc = av_pix_fmt_desc_get(s->pix_fmt);

560

1031  ff_thread_free(avctx);
1032 
1033  if (codec->priv_class && avctx->priv_data)
1034  av_opt_free(avctx->priv_data);
1035  av_opt_free(avctx);
1036 
1037  if (av_codec_is_encoder(avctx->codec)) {
1038 #if FF_API_CODED_FRAME
1039 FF_DISABLE_DEPRECATION_WARNINGS
1040  av_frame_free(&avctx->coded_frame);

1042 #endif
1043  av_freep(&avctx->extradata);

1049  av_freep(&avctx->subtitle_header);
1050 
1051  av_frame_free(&avci->to_free);
1052  av_frame_free(&avci->compat_decode_frame);
1053  av_frame_free(&avci->buffer_frame);
1054  av_packet_free(&avci->compat_encode_packet);

1057 
1058  av_packet_free(&avci->ds.in_pkt);

1060  av_bsf_free(&avci->bsf);
1061

1063  av_freep(&avci);

1065  avctx->codec = NULL;

1070 {

1072 
1073  if (av_codec_is_encoder(avctx->codec)) {
1074  int caps = avctx->codec->capabilities;
1075 
1076  if (!(caps & AV_CODEC_CAP_ENCODER_FLUSH)) {
1077  // Only encoders that explicitly declare support for it can be
1078  // flushed. Otherwise, this is a no-op.
1079  av_log(avctx, AV_LOG_WARNING, "Ignoring attempt to flush encoder "

1596 static int get_audio_frame_duration(enum AVCodecID id, int sr, int ch, int ba,
1597  uint32_t tag, int bits_per_coded_sample, int64_t bitrate,
1598  uint8_t * extradata, int frame_size, int frame_bytes)
1599 {
1600  int bps = av_get_exact_bits_per_sample(id);
1601  int framecount = (ba > 0 && frame_bytes / ba > 0) ? frame_bytes / ba : 1;
1602 
1603  /* codecs with an exact constant bits per sample */
1604  if (bps > 0 && ch > 0 && frame_bytes > 0 && ch < 32768 && bps < 32768)
1605  return (frame_bytes * 8LL) / (bps * ch);
1606  bps = bits_per_coded_sample;
1607 
1608  /* codecs with a fixed packet duration */
1609  switch (id) {
1610  case AV_CODEC_ID_ADPCM_ADX: return 32;
1611  case AV_CODEC_ID_ADPCM_IMA_QT: return 64;
1612  case AV_CODEC_ID_ADPCM_EA_XAS: return 128;
1613  case AV_CODEC_ID_AMR_NB:
1614  case AV_CODEC_ID_EVRC:
1615  case AV_CODEC_ID_GSM:
1616  case AV_CODEC_ID_QCELP:
1617  case AV_CODEC_ID_RA_288: return 160;
1618  case AV_CODEC_ID_AMR_WB:
1619  case AV_CODEC_ID_GSM_MS: return 320;
1620  case AV_CODEC_ID_MP1: return 384;
1621  case AV_CODEC_ID_ATRAC1: return 512;
1622  case AV_CODEC_ID_ATRAC9:
1623  case AV_CODEC_ID_ATRAC3:
1624  if (framecount > INT_MAX/1024)
1625  return 0;
1626  return 1024 * framecount;
1627  case AV_CODEC_ID_ATRAC3P: return 2048;
1628  case AV_CODEC_ID_MP2:
1629  case AV_CODEC_ID_MUSEPACK7: return 1152;
1630  case AV_CODEC_ID_AC3: return 1536;
1631  }
1632 
1633  if (sr > 0) {
1634  /* calc from sample rate */
1635  if (id == AV_CODEC_ID_TTA)
1636  return 256 * sr / 245;
1637  else if (id == AV_CODEC_ID_DST)
1638  return 588 * sr / 44100;
1639 
1640  if (ch > 0) {
1641  /* calc from sample rate and channels */
1642  if (id == AV_CODEC_ID_BINKAUDIO_DCT) {
1643  if (sr / 22050 > 22)
1644  return 0;
1645  return (480 << (sr / 22050)) / ch;
1646  }
1647  }
1648 
1649  if (id == AV_CODEC_ID_MP3)
1650  return sr <= 24000 ? 576 : 1152;
1651  }
1652 
1653  if (ba > 0) {
1654  /* calc from block_align */
1655  if (id == AV_CODEC_ID_SIPR) {
1656  switch (ba) {
1657  case 20: return 160;
1658  case 19: return 144;
1659  case 29: return 288;
1660  case 37: return 480;
1661  }
1662  } else if (id == AV_CODEC_ID_ILBC) {
1663  switch (ba) {
1664  case 38: return 160;
1665  case 50: return 240;
1666  }
1667  }
1668  }
1669 
1670  if (frame_bytes > 0) {
1671  /* calc from frame_bytes only */
1672  if (id == AV_CODEC_ID_TRUESPEECH)
1673  return 240 * (frame_bytes / 32);
1674  if (id == AV_CODEC_ID_NELLYMOSER)
1675  return 256 * (frame_bytes / 64);
1676  if (id == AV_CODEC_ID_RA_144)
1677  return 160 * (frame_bytes / 20);
1678 
1679  if (bps > 0) {
1680  /* calc from frame_bytes and bits_per_coded_sample */
1681  if (id == AV_CODEC_ID_ADPCM_G726 || id == AV_CODEC_ID_ADPCM_G726LE)
1682  return frame_bytes * 8 / bps;
1683  }
1684 
1685  if (ch > 0 && ch < INT_MAX/16) {
1686  /* calc from frame_bytes and channels */
1687  switch (id) {
1688  case AV_CODEC_ID_FASTAUDIO:
1689  return frame_bytes / (40 * ch) * 256;
1690  case AV_CODEC_ID_ADPCM_IMA_MOFLEX:
1691  return (frame_bytes - 4 * ch) / (128 * ch) * 256;
1692  case AV_CODEC_ID_ADPCM_AFC:
1693  return frame_bytes / (9 * ch) * 16;
1694  case AV_CODEC_ID_ADPCM_PSX:
1695  case AV_CODEC_ID_ADPCM_DTK:
1696  return frame_bytes / (16 * ch) * 28;
1697  case AV_CODEC_ID_ADPCM_4XM:
1698  case AV_CODEC_ID_ADPCM_IMA_DAT4:
1699  case AV_CODEC_ID_ADPCM_IMA_ISS:
1700  return (frame_bytes - 4 * ch) * 2 / ch;
1701  case AV_CODEC_ID_ADPCM_IMA_SMJPEG:
1702  return (frame_bytes - 4) * 2 / ch;
1703  case AV_CODEC_ID_ADPCM_IMA_AMV:
1704  return (frame_bytes - 8) * 2;
1705  case AV_CODEC_ID_ADPCM_THP:
1706  case AV_CODEC_ID_ADPCM_THP_LE:
1707  if (extradata)
1708  return frame_bytes * 14 / (8 * ch);
1709  break;
1710  case AV_CODEC_ID_ADPCM_XA:
1711  return (frame_bytes / 128) * 224 / ch;
1712  case AV_CODEC_ID_INTERPLAY_DPCM:
1713  return (frame_bytes - 6 - ch) / ch;
1714  case AV_CODEC_ID_ROQ_DPCM:
1715  return (frame_bytes - 8) / ch;
1716  case AV_CODEC_ID_XAN_DPCM:
1717  return (frame_bytes - 2 * ch) / ch;
1718  case AV_CODEC_ID_MACE3:
1719  return 3 * frame_bytes / ch;
1720  case AV_CODEC_ID_MACE6:
1721  return 6 * frame_bytes / ch;
1722  case AV_CODEC_ID_PCM_LXF:
1723  return 2 * (frame_bytes / (5 * ch));
1724  case AV_CODEC_ID_IAC:
1725  case AV_CODEC_ID_IMC:
1726  return 4 * frame_bytes / ch;
1727  }
1728 
1729  if (tag) {
1730  /* calc from frame_bytes, channels, and codec_tag */
1731  if (id == AV_CODEC_ID_SOL_DPCM) {
1732  if (tag == 3)
1733  return frame_bytes / ch;
1734  else
1735  return frame_bytes * 2 / ch;
1736  }
1737  }
1738 
1739  if (ba > 0) {
1740  /* calc from frame_bytes, channels, and block_align */
1741  int blocks = frame_bytes / ba;
1742  switch (id) {
1743  case AV_CODEC_ID_ADPCM_IMA_WAV:
1744  if (bps < 2 || bps > 5)
1745  return 0;
1746  return blocks * (1 + (ba - 4 * ch) / (bps * ch) * 8);
1747  case AV_CODEC_ID_ADPCM_IMA_DK3:
1748  return blocks * (((ba - 16) * 2 / 3 * 4) / ch);
1749  case AV_CODEC_ID_ADPCM_IMA_DK4:
1750  return blocks * (1 + (ba - 4 * ch) * 2 / ch);
1751  case AV_CODEC_ID_ADPCM_IMA_RAD:
1752  return blocks * ((ba - 4 * ch) * 2 / ch);
1753  case AV_CODEC_ID_ADPCM_MS:
1754  return blocks * (2 + (ba - 7 * ch) * 2 / ch);
1755  case AV_CODEC_ID_ADPCM_MTAF:
1756  return blocks * (ba - 16) * 2 / ch;
1757  }
1758  }
1759 
1760  if (bps > 0) {
1761  /* calc from frame_bytes, channels, and bits_per_coded_sample */
1762  switch (id) {
1763  case AV_CODEC_ID_PCM_DVD:
1764  if(bps<4 || frame_bytes<3)
1765  return 0;
1766  return 2 * ((frame_bytes - 3) / ((bps * 2 / 8) * ch));
1767  case AV_CODEC_ID_PCM_BLURAY:
1768  if(bps<4 || frame_bytes<4)
1769  return 0;
1770  return (frame_bytes - 4) / ((FFALIGN(ch, 2) * bps) / 8);
1771  case AV_CODEC_ID_S302M:
1772  return 2 * (frame_bytes / ((bps + 4) / 4)) / ch;
1773  }
1774  }
1775  }
1776  }
1777 
1778  /* Fall back on using frame_size */
1779  if (frame_size > 1 && frame_bytes)
1780  return frame_size;
1781 
1782  //For WMA we currently have no other means to calculate duration thus we
1783  //do it here by assuming CBR, which is true for all known cases.
1784  if (bitrate > 0 && frame_bytes > 0 && sr > 0 && ba > 1) {
1785  if (id == AV_CODEC_ID_WMAV1 || id == AV_CODEC_ID_WMAV2)
1786  return (frame_bytes * 8LL * sr) / bitrate;
1787  }
1788 
1789  return 0;
1790 }
