58 static inline int64_t bs_get_v(const uint8_t **bs)
59 {
60  uint64_t v = 0;
61  int br = 0;
62  int c;
63 
64  do {
65  c = **bs; (*bs)++;
66  v <<= 7;
67  v |= c & 0x7F;
68  br++;
69  if (br > 10)
70  return -1;
71  } while (c & 0x80);
72 
73  return v - br;
74 }

76 static int mpc8_probe(const AVProbeData *p)
77 {
78  const uint8_t *bs = p->buf + 4;
79  const uint8_t *bs_end = bs + p->buf_size;
80  int64_t size;
81 
82  if (p->buf_size < 16)
83  return 0;
84  if (AV_RL32(p->buf) != TAG_MPCK)
85  return 0;
86  while (bs < bs_end + 3) {
87  int header_found = (bs[0] == 'S' && bs[1] == 'H');
88  if (bs[0] < 'A' || bs[0] > 'Z' || bs[1] < 'A' || bs[1] > 'Z')
89  return 0;
90  bs += 2;
91  size = bs_get_v(&bs);
92  if (size < 2)
93  return 0;
94  if (size >= bs_end - bs + 2)
95  return AVPROBE_SCORE_EXTENSION - 1; // seems to be valid MPC but no header yet
96  if (header_found) {
97  if (size < 11 || size > 28)
98  return 0;
99  if (!AV_RL32(bs)) //zero CRC is invalid
100  return 0;
101  return AVPROBE_SCORE_MAX;
102  } else {
103  bs += size - 2;
104  }
105  }
106  return 0;
107 }

109 static inline int64_t gb_get_v(GetBitContext *gb)
110 {
111  uint64_t v = 0;
112  int bits = 0;
113  while(get_bits1(gb) && bits < 64-7){
114  v <<= 7;
115  v |= get_bits(gb, 7);
116  bits += 7;
117  }
118  v <<= 7;
119  v |= get_bits(gb, 7);
120 
121  return v;
122 }

133 static void mpc8_parse_seektable(AVFormatContext *s, int64_t off)
134 {
135  MPCContext *c = s->priv_data;
136  int tag;
137  int64_t size, pos, ppos[2];
138  uint8_t *buf;
139  int i, t, seekd, ret;
140  GetBitContext gb;
141 
142  if (s->nb_streams == 0) {
143  av_log(s, AV_LOG_ERROR, "No stream added before parsing seek table\n");
144  return;
145  }
146 
147  avio_seek(s->pb, off, SEEK_SET);
148  mpc8_get_chunk_header(s->pb, &tag, &size);
149  if(tag != TAG_SEEKTABLE){
150  av_log(s, AV_LOG_ERROR, "No seek table at given position\n");
151  return;
152  }
153  if (size > INT_MAX/10 || size<=0) {
154  av_log(s, AV_LOG_ERROR, "Bad seek table size\n");
155  return;
156  }
157  if(!(buf = av_malloc(size + AV_INPUT_BUFFER_PADDING_SIZE)))
158  return;
159  ret = avio_read(s->pb, buf, size);
160  if (ret != size) {
161  av_log(s, AV_LOG_ERROR, "seek table truncated\n");
162  av_free(buf);
163  return;
164  }
165  memset(buf+size, 0, AV_INPUT_BUFFER_PADDING_SIZE);
166 
167  init_get_bits(&gb, buf, size * 8);
168  size = gb_get_v(&gb);
169  if(size > UINT_MAX/4 || size > c->samples/1152){
170  av_log(s, AV_LOG_ERROR, "Seek table is too big\n");
171  av_free(buf);
172  return;
173  }
174  seekd = get_bits(&gb, 4);
175  for(i = 0; i < 2; i++){
176  pos = gb_get_v(&gb) + c->header_pos;
177  ppos[1 - i] = pos;
178  av_add_index_entry(s->streams[0], pos, i, 0, 0, AVINDEX_KEYFRAME);
179  }
180  for(; i < size; i++){
181  t = get_unary(&gb, 1, 33) << 12;
182  t += get_bits(&gb, 12);
183  if(t & 1)
184  t = -(t & ~1);
185  pos = (t >> 1) + ppos[0]*2 - ppos[1];
186  av_add_index_entry(s->streams[0], pos, i << seekd, 0, 0, AVINDEX_KEYFRAME);
187  ppos[1] = ppos[0];
188  ppos[0] = pos;
189  }
190  av_free(buf);
191 }

193 static void mpc8_handle_chunk(AVFormatContext *s, int tag, int64_t chunk_pos, int64_t size)
194 {
195  AVIOContext *pb = s->pb;
196  int64_t pos, off;
197 
198  switch(tag){
199  case TAG_SEEKTBLOFF:
200  pos = avio_tell(pb) + size;
201  off = ffio_read_varlen(pb);
202  mpc8_parse_seektable(s, chunk_pos + off);
203  avio_seek(pb, pos, SEEK_SET);
204  break;
205  default:
206  avio_skip(pb, size);
207  }

210 static int mpc8_read_header(AVFormatContext *s)
211 {
212  MPCContext *c = s->priv_data;
213  AVIOContext *pb = s->pb;
214  AVStream *st;
215  int tag = 0, ret;
216  int64_t size, pos;
217 
218  c->header_pos = avio_tell(pb);
219  if(avio_rl32(pb) != TAG_MPCK){
220  av_log(s, AV_LOG_ERROR, "Not a Musepack8 file\n");
221  return AVERROR_INVALIDDATA;
222  }
223 
224  while(!avio_feof(pb)){
225  pos = avio_tell(pb);
226  mpc8_get_chunk_header(pb, &tag, &size);
227  if (size < 0) {
228  av_log(s, AV_LOG_ERROR, "Invalid chunk length\n");
229  return AVERROR_INVALIDDATA;
230  }
231  if(tag == TAG_STREAMHDR)
232  break;
233  mpc8_handle_chunk(s, tag, pos, size);
234  }
235  if(tag != TAG_STREAMHDR){
236  av_log(s, AV_LOG_ERROR, "Stream header not found\n");
237  return AVERROR_INVALIDDATA;
238  }
239  pos = avio_tell(pb);
240  avio_skip(pb, 4); //CRC
241  c->ver = avio_r8(pb);
242  if(c->ver != 8){
243  avpriv_report_missing_feature(s, "Stream version %d", c->ver);
244  return AVERROR_PATCHWELCOME;
245  }
246  c->samples = ffio_read_varlen(pb);
247  ffio_read_varlen(pb); //silence samples at the beginning
248 
249  st = avformat_new_stream(s, NULL);
250  if (!st)
251  return AVERROR(ENOMEM);
252  st->codecpar->codec_type = AVMEDIA_TYPE_AUDIO;
253  st->codecpar->codec_id = AV_CODEC_ID_MUSEPACK8;
254  st->codecpar->bits_per_coded_sample = 16;
255 
256  if ((ret = ff_get_extradata(s, st->codecpar, pb, 2)) < 0)
257  return ret;
258 
259  st->codecpar->channels = (st->codecpar->extradata[1] >> 4) + 1;
260  st->codecpar->sample_rate = mpc8_rate[st->codecpar->extradata[0] >> 5];
261  avpriv_set_pts_info(st, 32, 1152 << (st->codecpar->extradata[1]&3)*2, st->codecpar->sample_rate);
262  st->start_time = 0;
263  st->duration = c->samples / (1152 << (st->codecpar->extradata[1]&3)*2);
264  size -= avio_tell(pb) - pos;
265  if (size > 0)
266  avio_skip(pb, size);
267 
268  if (pb->seekable & AVIO_SEEKABLE_NORMAL) {
269  int64_t pos = avio_tell(s->pb);
270  c->apetag_start = ff_ape_parse_tag(s);
271  avio_seek(s->pb, pos, SEEK_SET);
272  }
273 
274  return 0;
275 }

277 static int mpc8_read_packet(AVFormatContext *s, AVPacket *pkt)
278 {
279  MPCContext *c = s->priv_data;
280  int tag, ret;
281  int64_t pos, size;
282 
283  while(!avio_feof(s->pb)){
284  pos = avio_tell(s->pb);
285 
286  /* don't return bogus packets with the ape tag data */
287  if (c->apetag_start && pos >= c->apetag_start)
288  return AVERROR_EOF;
289 
290  mpc8_get_chunk_header(s->pb, &tag, &size);
291  if (size < 0)
292  return -1;
293  if(tag == TAG_AUDIOPACKET){
294  if ((ret = av_get_packet(s->pb, pkt, size)) < 0)
295  return ret;
296  pkt->stream_index = 0;
297  pkt->duration = 1;
298  return 0;
299  }
300  if(tag == TAG_STREAMEND)
301  return AVERROR_EOF;
302  mpc8_handle_chunk(s, tag, pos, size);
303  }
304  return AVERROR_EOF;
305 }
