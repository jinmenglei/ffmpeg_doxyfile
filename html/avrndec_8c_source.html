35 static av_cold int init(AVCodecContext *avctx)
36 {
37  AVRnContext *a = avctx->priv_data;
38  int ret;
39 
40  // Support "Resolution 1:1" for Avid AVI Codec
41  a->is_mjpeg = avctx->extradata_size < 31 || memcmp(&avctx->extradata[28], "1:1", 3);
42 
43  if(!a->is_mjpeg && avctx->lowres) {
44  av_log(avctx, AV_LOG_ERROR, "lowres is not possible with rawvideo\n");
45  return AVERROR(EINVAL);
46  }
47 
48  if(a->is_mjpeg) {
49  const AVCodec *codec = avcodec_find_decoder(AV_CODEC_ID_MJPEG);
50  AVDictionary *thread_opt = NULL;
51  if (!codec) {
52  av_log(avctx, AV_LOG_ERROR, "MJPEG codec not found\n");
53  return AVERROR_DECODER_NOT_FOUND;
54  }
55 
56  a->mjpeg_avctx = avcodec_alloc_context3(codec);
57  if (!a->mjpeg_avctx)
58  return AVERROR(ENOMEM);
59 
60  av_dict_set(&thread_opt, "threads", "1", 0); // Is this needed ?
61  a->mjpeg_avctx->refcounted_frames = 1;
62  a->mjpeg_avctx->flags = avctx->flags;
63  a->mjpeg_avctx->idct_algo = avctx->idct_algo;
64  a->mjpeg_avctx->lowres = avctx->lowres;
65  a->mjpeg_avctx->width = avctx->width;
66  a->mjpeg_avctx->height = avctx->height;
67 
68  if ((ret = ff_codec_open2_recursive(a->mjpeg_avctx, codec, &thread_opt)) < 0) {
69  av_log(avctx, AV_LOG_ERROR, "MJPEG codec failed to open\n");
70  }
71  av_dict_free(&thread_opt);
72 
73  return ret;
74  }
75 
76  if ((ret = av_image_check_size(avctx->width, avctx->height, 0, avctx)) < 0)
77  return ret;
78 
79  avctx->pix_fmt = AV_PIX_FMT_UYVY422;
80 
81  if(avctx->extradata_size >= 9 && avctx->extradata[4]+28 < avctx->extradata_size) {
82  int ndx = avctx->extradata[4] + 4;
83  a->interlace = !memcmp(avctx->extradata + ndx, "1:1(", 4);
84  if(a->interlace) {
85  a->tff = avctx->extradata[ndx + 24] == 1;
86  }
87  }
88 
89  return 0;
90 }

101 static int decode_frame(AVCodecContext *avctx, void *data,
102  int *got_frame, AVPacket *avpkt)
103 {
104  AVRnContext *a = avctx->priv_data;
105  AVFrame *p = data;
106  const uint8_t *buf = avpkt->data;
107  int buf_size = avpkt->size;
108  int y, ret, true_height;
109 
110  if(a->is_mjpeg) {
111  ret = avcodec_decode_video2(a->mjpeg_avctx, data, got_frame, avpkt);
112 
113  if (ret >= 0 && *got_frame && avctx->width <= p->width && avctx->height <= p->height) {
114  int shift = p->height - avctx->height;
115  int subsample_h, subsample_v;
116 
117  av_pix_fmt_get_chroma_sub_sample(p->format, &subsample_h, &subsample_v);
118 
119  p->data[0] += p->linesize[0] * shift;
120  if (p->data[2]) {
121  p->data[1] += p->linesize[1] * (shift>>subsample_v);
122  p->data[2] += p->linesize[2] * (shift>>subsample_v);
123  }
124 
125  p->width = avctx->width;
126  p->height = avctx->height;
127  }
128  avctx->pix_fmt = a->mjpeg_avctx->pix_fmt;
129  return ret;
130  }
131 
132  true_height = buf_size / (2*avctx->width);
133 
134  if(buf_size < 2*avctx->width * avctx->height) {
135  av_log(avctx, AV_LOG_ERROR, "packet too small\n");
136  return AVERROR_INVALIDDATA;
137  }
138 
139  if ((ret = ff_get_buffer(avctx, p, 0)) < 0)
140  return ret;
141  p->pict_type= AV_PICTURE_TYPE_I;
142  p->key_frame= 1;
143 
144  if(a->interlace) {
145  buf += (true_height - avctx->height)*avctx->width;
146  for(y = 0; y < avctx->height-1; y+=2) {
147  memcpy(p->data[0] + (y+ a->tff)*p->linesize[0], buf , 2*avctx->width);
148  memcpy(p->data[0] + (y+!a->tff)*p->linesize[0], buf + avctx->width*true_height+4, 2*avctx->width);
149  buf += 2*avctx->width;
150  }
151  } else {
152  buf += (true_height - avctx->height)*avctx->width*2;
153  for(y = 0; y < avctx->height; y++) {
154  memcpy(p->data[0] + y*p->linesize[0], buf, 2*avctx->width);
155  buf += 2*avctx->width;
156  }
157  }
158 
159  *got_frame = 1;
160  return buf_size;
161 }
