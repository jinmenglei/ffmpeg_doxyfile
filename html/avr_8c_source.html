27 static int avr_probe(const AVProbeData *p)
28 {
29  if (AV_RL32(p->buf) != MKTAG('2', 'B', 'I', 'T'))
30  return 0;
31 
32  if (!AV_RB16(p->buf+12) || AV_RB16(p->buf+12) > 256) // channels
33  return AVPROBE_SCORE_EXTENSION/2;
34  if (AV_RB16(p->buf+14) > 256) // bps
35  return AVPROBE_SCORE_EXTENSION/2;
36 
37  return AVPROBE_SCORE_EXTENSION;
38 }

40 static int avr_read_header(AVFormatContext *s)
41 {
42  uint16_t chan, sign, bps;
43  AVStream *st;
44 
45  st = avformat_new_stream(s, NULL);
46  if (!st)
47  return AVERROR(ENOMEM);
48 
49  st->codecpar->codec_type = AVMEDIA_TYPE_AUDIO;
50 
51  avio_skip(s->pb, 4); // magic
52  avio_skip(s->pb, 8); // sample_name
53 
54  chan = avio_rb16(s->pb);
55  if (!chan) {
56  st->codecpar->channels = 1;
57  } else if (chan == 0xFFFFu) {
58  st->codecpar->channels = 2;
59  } else {
60  avpriv_request_sample(s, "chan %d", chan);
61  return AVERROR_PATCHWELCOME;
62  }
63 
64  st->codecpar->bits_per_coded_sample = bps = avio_rb16(s->pb);
65 
66  sign = avio_rb16(s->pb);
67 
68  avio_skip(s->pb, 2); // loop
69  avio_skip(s->pb, 2); // midi
70  avio_skip(s->pb, 1); // replay speed
71 
72  st->codecpar->sample_rate = avio_rb24(s->pb);
73  avio_skip(s->pb, 4 * 3);
74  avio_skip(s->pb, 2 * 3);
75  avio_skip(s->pb, 20);
76  avio_skip(s->pb, 64);
77 
78  st->codecpar->codec_id = ff_get_pcm_codec_id(bps, 0, 1, sign);
79  if (st->codecpar->codec_id == AV_CODEC_ID_NONE) {
80  avpriv_request_sample(s, "Bps %d and sign %d", bps, sign);
81  return AVERROR_PATCHWELCOME;
82  }
83 
84  st->codecpar->block_align = bps * st->codecpar->channels / 8;
85 
86  avpriv_set_pts_info(st, 64, 1, st->codecpar->sample_rate);
87  return 0;
88 }
