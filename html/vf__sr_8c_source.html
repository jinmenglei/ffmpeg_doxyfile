65 static av_cold int init(AVFilterContext *context)
66 {
67  SRContext *sr_context = context->priv;
68 
69  sr_context->dnn_module = ff_get_dnn_module(sr_context->backend_type);
70  if (!sr_context->dnn_module){
71  av_log(context, AV_LOG_ERROR, "could not create DNN module for requested backend\n");
72  return AVERROR(ENOMEM);
73  }
74 
75  if (!sr_context->model_filename){
76  av_log(context, AV_LOG_ERROR, "model file for network was not specified\n");
77  return AVERROR(EIO);
78  }
79  if (!sr_context->dnn_module->load_model) {
80  av_log(context, AV_LOG_ERROR, "load_model for network was not specified\n");
81  return AVERROR(EIO);
82  }
83  sr_context->model = (sr_context->dnn_module->load_model)(sr_context->model_filename, NULL, NULL);
84  if (!sr_context->model){
85  av_log(context, AV_LOG_ERROR, "could not load DNN model\n");
86  return AVERROR(EIO);
87  }
88 
89  return 0;
90 }

108 static int config_output(AVFilterLink *outlink)
109 {
110  AVFilterContext *context = outlink->src;
111  SRContext *ctx = context->priv;
112  DNNReturnType result;
113  AVFilterLink *inlink = context->inputs[0];
114  int out_width, out_height;
115 
116  // have a try run in case that the dnn model resize the frame
117  result = ctx->model->get_output(ctx->model->model, "x", inlink->w, inlink->h,
118  "y", &out_width, &out_height);
119  if (result != DNN_SUCCESS) {
120  av_log(ctx, AV_LOG_ERROR, "could not get output from the model\n");
121  return AVERROR(EIO);
122  }
123 
124  if (inlink->w != out_width || inlink->h != out_height) {
125  //espcn
126  outlink->w = out_width;
127  outlink->h = out_height;
128  if (inlink->format != AV_PIX_FMT_GRAY8){
129  const AVPixFmtDescriptor *desc = av_pix_fmt_desc_get(inlink->format);
130  int sws_src_h = AV_CEIL_RSHIFT(inlink->h, desc->log2_chroma_h);
131  int sws_src_w = AV_CEIL_RSHIFT(inlink->w, desc->log2_chroma_w);
132  int sws_dst_h = AV_CEIL_RSHIFT(outlink->h, desc->log2_chroma_h);
133  int sws_dst_w = AV_CEIL_RSHIFT(outlink->w, desc->log2_chroma_w);
134  ctx->sws_uv_scale = sws_getContext(sws_src_w, sws_src_h, AV_PIX_FMT_GRAY8,
135  sws_dst_w, sws_dst_h, AV_PIX_FMT_GRAY8,
136  SWS_BICUBIC, NULL, NULL, NULL);
137  ctx->sws_uv_height = sws_src_h;
138  }
139  } else {
140  //srcnn
141  outlink->w = out_width * ctx->scale_factor;
142  outlink->h = out_height * ctx->scale_factor;
143  ctx->sws_pre_scale = sws_getContext(inlink->w, inlink->h, inlink->format,
144  outlink->w, outlink->h, outlink->format,
145  SWS_BICUBIC, NULL, NULL, NULL);
146  }
147 
148  return 0;
149 }

151 static int filter_frame(AVFilterLink *inlink, AVFrame *in)
152 {
153  AVFilterContext *context = inlink->dst;
154  SRContext *ctx = context->priv;
155  AVFilterLink *outlink = context->outputs[0];
156  AVFrame *out = ff_get_video_buffer(outlink, outlink->w, outlink->h);
157  DNNReturnType dnn_result;
158  const char *model_output_name = "y";
159 
160  if (!out){
161  av_log(context, AV_LOG_ERROR, "could not allocate memory for output frame\n");
162  av_frame_free(&in);
163  return AVERROR(ENOMEM);
164  }
165  av_frame_copy_props(out, in);
166 
167  if (ctx->sws_pre_scale) {
168  sws_scale(ctx->sws_pre_scale,
169  (const uint8_t **)in->data, in->linesize, 0, in->height,
170  out->data, out->linesize);
171  dnn_result = (ctx->dnn_module->execute_model)(ctx->model, "x", out,
172  (const char **)&model_output_name, 1, out);
173  } else {
174  dnn_result = (ctx->dnn_module->execute_model)(ctx->model, "x", in,
175  (const char **)&model_output_name, 1, out);
176  }
177 
178  if (dnn_result != DNN_SUCCESS){
179  av_log(ctx, AV_LOG_ERROR, "failed to execute loaded model\n");
180  av_frame_free(&in);
181  av_frame_free(&out);
182  return AVERROR(EIO);
183  }
184 
185  if (ctx->sws_uv_scale) {
186  sws_scale(ctx->sws_uv_scale, (const uint8_t **)(in->data + 1), in->linesize + 1,
187  0, ctx->sws_uv_height, out->data + 1, out->linesize + 1);
188  sws_scale(ctx->sws_uv_scale, (const uint8_t **)(in->data + 2), in->linesize + 2,
189  0, ctx->sws_uv_height, out->data + 2, out->linesize + 2);
190  }
191 
192  av_frame_free(&in);
193  return ff_filter_frame(outlink, out);
194 }
