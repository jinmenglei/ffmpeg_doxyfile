46 {

51 
52  declare_func_emms(AV_CPU_FLAG_MMX, void, uint8_t *dst, int16_t *res, ptrdiff_t stride);
53 
54  randomize_buffers(res0, size);
55  randomize_buffers2(dst0, size);
56  if (overflow_test)
57  res0[0] = 0x8000;

63  if (memcmp(dst0, dst1, size))

66 }
