27 static int probe(const AVProbeData *p)
28 {
29  if (p->buf_size < 2096)
30  return 0;
31  if ( AV_RL32(p->buf ) != 1
32  || AV_RL32(p->buf + 8) > 100000
33  || AV_RL32(p->buf + 12) > 8
34  || AV_RL32(p->buf + 16) != 2096
35  ||!AV_RL32(p->buf + 21)
36  || AV_RL16(p->buf + 25) != 2096
37  || AV_RL32(p->buf + 48) % AV_RL32(p->buf + 21)
38  )
39  return 0;
40  return AVPROBE_SCORE_EXTENSION;

44 static int read_header(AVFormatContext *s)
45 {
46  AVStream *st = avformat_new_stream(s, NULL);
47  if (!st)
48  return AVERROR(ENOMEM);
49 
50  st->codecpar->codec_type = AVMEDIA_TYPE_AUDIO;
51  st->codecpar->codec_id = AV_CODEC_ID_ADPCM_MS;
52 
53  avio_rl32(s->pb);
54  avio_rl32(s->pb);
55  st->codecpar->sample_rate = avio_rl32(s->pb);
56  st->codecpar->channels = avio_rl32(s->pb);
57  if (st->codecpar->channels > FF_SANE_NB_CHANNELS || st->codecpar->channels <= 0)
58  return AVERROR(ENOSYS);
59  s->internal->data_offset = avio_rl32(s->pb);
60  avio_r8(s->pb);
61  st->codecpar->block_align = avio_rl32(s->pb);
62  if (st->codecpar->block_align > INT_MAX / FF_SANE_NB_CHANNELS || st->codecpar->block_align <= 0)
63  return AVERROR_INVALIDDATA;
64  st->codecpar->block_align *= st->codecpar->channels;
65 
66  avio_seek(s->pb, s->internal->data_offset, SEEK_SET);
67 
68  return 0;
69 }
