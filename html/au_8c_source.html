67 static int au_read_annotation(AVFormatContext *s, int size)
68 {
69  static const char keys[][7] = {
70  "title",
71  "artist",
72  "album",
73  "track",
74  "genre",
75  };
76  AVIOContext *pb = s->pb;
77  enum { PARSE_KEY, PARSE_VALUE, PARSE_FINISHED } state = PARSE_KEY;
78  char c;
79  AVBPrint bprint;
80  char * key = NULL;
81  char * value = NULL;
82  int ret, i;
83 
84  av_bprint_init(&bprint, 64, AV_BPRINT_SIZE_UNLIMITED);
85 
86  while (size-- > 0) {
87  if (avio_feof(pb)) {
88  av_bprint_finalize(&bprint, NULL);
89  av_freep(&key);
90  return AVERROR_EOF;
91  }
92  c = avio_r8(pb);
93  switch(state) {
94  case PARSE_KEY:
95  if (c == '\0') {
96  state = PARSE_FINISHED;
97  } else if (c == '=') {
98  ret = av_bprint_finalize(&bprint, &key);
99  if (ret < 0)
100  return ret;
101  av_bprint_init(&bprint, 64, AV_BPRINT_SIZE_UNLIMITED);
102  state = PARSE_VALUE;
103  } else {
104  av_bprint_chars(&bprint, c, 1);
105  }
106  break;
107  case PARSE_VALUE:
108  if (c == '\0' || c == '\n') {
109  if (av_bprint_finalize(&bprint, &value) != 0) {
110  av_log(s, AV_LOG_ERROR, "Memory error while parsing AU metadata.\n");
111  } else {
112  av_bprint_init(&bprint, 64, AV_BPRINT_SIZE_UNLIMITED);
113  for (i = 0; i < FF_ARRAY_ELEMS(keys); i++) {
114  if (av_strcasecmp(keys[i], key) == 0) {
115  av_dict_set(&(s->metadata), keys[i], value, AV_DICT_DONT_STRDUP_VAL);
116  value = NULL;
117  break;
118  }
119  }
120  }
121  av_freep(&key);
122  av_freep(&value);
123  state = (c == '\0') ? PARSE_FINISHED : PARSE_KEY;
124  } else {
125  av_bprint_chars(&bprint, c, 1);
126  }
127  break;
128  case PARSE_FINISHED:
129  break;
130  default:
131  /* should never happen */
132  av_assert0(0);
133  }
134  }
135  av_bprint_finalize(&bprint, NULL);
136  av_freep(&key);
137  return 0;
138 }

142 static int au_read_header(AVFormatContext *s)
143 {
144  int size, data_size = 0;
145  unsigned int tag;
146  AVIOContext *pb = s->pb;
147  unsigned int id, channels, rate;
148  int bps, ba = 0;
149  enum AVCodecID codec;
150  AVStream *st;
151  int ret;
152 
153  tag = avio_rl32(pb);
154  if (tag != MKTAG('.', 's', 'n', 'd'))
155  return AVERROR_INVALIDDATA;
156  size = avio_rb32(pb); /* header size */
157  data_size = avio_rb32(pb); /* data size in bytes */
158 
159  if (data_size < 0 && data_size != AU_UNKNOWN_SIZE) {
160  av_log(s, AV_LOG_ERROR, "Invalid negative data size '%d' found\n", data_size);
161  return AVERROR_INVALIDDATA;
162  }
163 
164  id = avio_rb32(pb);
165  rate = avio_rb32(pb);
166  channels = avio_rb32(pb);
167 
168  if (size > 24) {
169  /* parse annotation field to get metadata */
170  ret = au_read_annotation(s, size - 24);
171  if (ret < 0)
172  return ret;
173  }
174 
175  codec = ff_codec_get_id(codec_au_tags, id);
176 
177  if (codec == AV_CODEC_ID_NONE) {
178  avpriv_request_sample(s, "unknown or unsupported codec tag: %u", id);
179  return AVERROR_PATCHWELCOME;
180  }
181 
182  bps = av_get_bits_per_sample(codec);
183  if (codec == AV_CODEC_ID_ADPCM_G726LE) {
184  if (id == MKBETAG('7','2','6','2')) {
185  bps = 2;
186  } else {
187  const uint8_t bpcss[] = {4, 0, 3, 5};
188  av_assert0(id >= 23 && id < 23 + 4);
189  ba = bpcss[id - 23];
190  bps = bpcss[id - 23];
191  }
192  } else if (!bps) {
193  avpriv_request_sample(s, "Unknown bits per sample");
194  return AVERROR_PATCHWELCOME;
195  }
196 
197  if (channels == 0 || channels >= INT_MAX / (BLOCK_SIZE * bps >> 3)) {
198  av_log(s, AV_LOG_ERROR, "Invalid number of channels %u\n", channels);
199  return AVERROR_INVALIDDATA;
200  }
201 
202  if (rate == 0 || rate > INT_MAX) {
203  av_log(s, AV_LOG_ERROR, "Invalid sample rate: %u\n", rate);
204  return AVERROR_INVALIDDATA;
205  }
206 
207  st = avformat_new_stream(s, NULL);
208  if (!st)
209  return AVERROR(ENOMEM);
210  st->codecpar->codec_type = AVMEDIA_TYPE_AUDIO;
211  st->codecpar->codec_tag = id;
212  st->codecpar->codec_id = codec;
213  st->codecpar->channels = channels;
214  st->codecpar->sample_rate = rate;
215  st->codecpar->bits_per_coded_sample = bps;
216  st->codecpar->bit_rate = channels * rate * bps;
217  st->codecpar->block_align = ba ? ba : FFMAX(bps * st->codecpar->channels / 8, 1);
218  if (data_size != AU_UNKNOWN_SIZE)
219  st->duration = (((int64_t)data_size)<<3) / (st->codecpar->channels * (int64_t)bps);
220 
221  st->start_time = 0;
222  avpriv_set_pts_info(st, 64, 1, rate);
223 
224  return 0;
225 }

247 static int au_get_annotations(AVFormatContext *s, AVBPrint *annotations)
248 {
249  static const char keys[][7] = {
250  "Title",
251  "Artist",
252  "Album",
253  "Track",
254  "Genre",
255  };
256  int cnt = 0;
257  AVDictionary *m = s->metadata;
258  AVDictionaryEntry *t = NULL;
259 
260  for (int i = 0; i < FF_ARRAY_ELEMS(keys); i++) {
261  t = av_dict_get(m, keys[i], NULL, 0);
262  if (t != NULL) {
263  if (cnt++)
264  av_bprint_chars(annotations, '\n', 1);
265  av_bprintf(annotations, "%s=%s", keys[i], t->value);
266  }
267  }
268  /* The specification requires the annotation field to be zero-terminated
269  * and its length to be a multiple of eight, so pad with 0's */
270  av_bprint_chars(annotations, '\0', 8);
271  return av_bprint_is_complete(annotations) ? 0 : AVERROR(ENOMEM);
272 }

274 static int au_write_header(AVFormatContext *s)
275 {
276  int ret;
277  AUContext *au = s->priv_data;
278  AVIOContext *pb = s->pb;
279  AVCodecParameters *par = s->streams[0]->codecpar;
280  AVBPrint annotations;
281 
282  if (s->nb_streams != 1) {
283  av_log(s, AV_LOG_ERROR, "only one stream is supported\n");
284  return AVERROR(EINVAL);
285  }
286 
287  par->codec_tag = ff_codec_get_tag(codec_au_tags, par->codec_id);
288  if (!par->codec_tag) {
289  av_log(s, AV_LOG_ERROR, "unsupported codec\n");
290  return AVERROR(EINVAL);
291  }
292 
293  av_bprint_init(&annotations, 0, INT_MAX - 24);
294  ret = au_get_annotations(s, &annotations);
295  if (ret < 0)
296  goto fail;
297  au->header_size = 24 + annotations.len & ~7;
298 
299  ffio_wfourcc(pb, ".snd"); /* magic number */
300  avio_wb32(pb, au->header_size); /* header size */
301  avio_wb32(pb, AU_UNKNOWN_SIZE); /* data size */
302  avio_wb32(pb, par->codec_tag); /* codec ID */
303  avio_wb32(pb, par->sample_rate);
304  avio_wb32(pb, par->channels);
305  avio_write(pb, annotations.str, annotations.len & ~7);
306 
307 fail:
308  av_bprint_finalize(&annotations, NULL);
309 
310  return ret;
311 }

313 static int au_write_trailer(AVFormatContext *s)
314 {
315  AVIOContext *pb = s->pb;
316  AUContext *au = s->priv_data;
317  int64_t file_size = avio_tell(pb);
318 
319  if ((s->pb->seekable & AVIO_SEEKABLE_NORMAL) && file_size < INT32_MAX) {
320  /* update file size */
321  avio_seek(pb, 8, SEEK_SET);
322  avio_wb32(pb, (uint32_t)(file_size - au->header_size));
323  avio_seek(pb, file_size, SEEK_SET);
324  }
325 
326  return 0;
327 }
