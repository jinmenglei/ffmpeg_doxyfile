127 void ff_mjpeg_build_huffman_codes(uint8_t *huff_size, uint16_t *huff_code,
128  const uint8_t *bits_table,
129  const uint8_t *val_table)
130 {
131  int i, j, k,nb, code, sym;
132 
133  k = 0;
134  code = 0;
135  for(i=1;i<=16;i++) {
136  nb = bits_table[i];
137  for(j=0;j<nb;j++) {
138  sym = val_table[k++];
139  huff_size[sym] = i;
140  huff_code[sym] = code;
141  code++;
142  }
143  code <<= 1;
144  }
145 }
