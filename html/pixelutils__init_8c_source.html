52 void ff_pixelutils_sad_init_x86(av_pixelutils_sad_fn *sad, int aligned)
53 {
54  int cpu_flags = av_get_cpu_flags();
55 
56  if (EXTERNAL_MMX(cpu_flags)) {
57  sad[2] = ff_pixelutils_sad_8x8_mmx;
58  }
59 
60  // The best way to use SSE2 would be to do 2 SADs in parallel,
61  // but we'd have to modify the pixelutils API to return SIMD functions.
62 
63  // It's probably not faster to shuffle data around
64  // to get two lines of 8 pixels into a single 16byte register,
65  // so just use the MMX 8x8 version even when SSE2 is available.
66  if (EXTERNAL_MMXEXT(cpu_flags)) {
67  sad[2] = ff_pixelutils_sad_8x8_mmxext;
68  sad[3] = ff_pixelutils_sad_16x16_mmxext;
69  }
70 
71  if (EXTERNAL_SSE2(cpu_flags)) {
72  switch (aligned) {
73  case 0: sad[3] = ff_pixelutils_sad_16x16_sse2; break; // src1 unaligned, src2 unaligned
74  case 1: sad[3] = ff_pixelutils_sad_u_16x16_sse2; break; // src1 aligned, src2 unaligned
75  case 2: sad[3] = ff_pixelutils_sad_a_16x16_sse2; break; // src1 aligned, src2 aligned
76  }
77  }
78 
79  if (EXTERNAL_SSE2(cpu_flags)) {
80  switch (aligned) {
81  case 0: sad[4] = ff_pixelutils_sad_32x32_sse2; break; // src1 unaligned, src2 unaligned
82  case 1: sad[4] = ff_pixelutils_sad_u_32x32_sse2; break; // src1 aligned, src2 unaligned
83  case 2: sad[4] = ff_pixelutils_sad_a_32x32_sse2; break; // src1 aligned, src2 aligned
84  }
85  }
86 
87  if (EXTERNAL_AVX2_FAST(cpu_flags)) {
88  switch (aligned) {
89  case 0: sad[4] = ff_pixelutils_sad_32x32_avx2; break; // src1 unaligned, src2 unaligned
90  case 1: sad[4] = ff_pixelutils_sad_u_32x32_avx2; break; // src1 aligned, src2 unaligned
91  case 2: sad[4] = ff_pixelutils_sad_a_32x32_avx2; break; // src1 aligned, src2 aligned
92  }
93  }
94 }
