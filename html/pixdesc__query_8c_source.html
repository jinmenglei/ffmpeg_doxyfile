57 int main(void)
58 {
59  int i, j;
60 
61  for (i = 0; i < FF_ARRAY_ELEMS(query_tab); i++) {
62  const char **pix_fmts = NULL;
63  int nb_pix_fmts = 0;
64  const AVPixFmtDescriptor *pix_desc = NULL;
65 
66  while ((pix_desc = av_pix_fmt_desc_next(pix_desc))) {
67  enum AVPixelFormat pix_fmt = av_pix_fmt_desc_get_id(pix_desc);
68  if (query_tab[i].cond(pix_fmt)) {
69  const char *pix_name = pix_desc->name;
70  if (pix_fmt == AV_PIX_FMT_RGB32) pix_name = "rgb32";
71  else if (pix_fmt == AV_PIX_FMT_RGB32_1) pix_name = "rgb32_1";
72  else if (pix_fmt == AV_PIX_FMT_BGR32) pix_name = "bgr32";
73  else if (pix_fmt == AV_PIX_FMT_BGR32_1) pix_name = "bgr32_1";
74 
75  av_dynarray_add(&pix_fmts, &nb_pix_fmts, (void *)pix_name);
76  }
77  }
78 
79  if (pix_fmts) {
80  qsort(pix_fmts, nb_pix_fmts, sizeof(*pix_fmts), cmp_str);
81 
82  printf("%s:\n", query_tab[i].class);

84  printf(" %s\n", pix_fmts[j]);
85  printf("\n");

88  }
89  }
