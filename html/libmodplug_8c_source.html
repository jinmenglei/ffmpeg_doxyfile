133 static int modplug_load_metadata(AVFormatContext *s)
134 {
135  ModPlugContext *modplug = s->priv_data;
136  ModPlugFile *f = modplug->f;
137  char *extra;
138  const char *name = ModPlug_GetName(f);
139  const char *msg = ModPlug_GetMessage(f);
140 
141  unsigned n_instruments = ModPlug_NumInstruments(f);
142  unsigned n_samples = ModPlug_NumSamples(f);
143  unsigned n_patterns = ModPlug_NumPatterns(f);
144  unsigned n_channels = ModPlug_NumChannels(f);
145 
146  if (name && *name) av_dict_set(&s->metadata, "name", name, 0);
147  if (msg && *msg) av_dict_set(&s->metadata, "message", msg, 0);
148 
149  extra = av_asprintf("%u pattern%s, %u channel%s",
150  n_patterns, n_patterns > 1 ? "s" : "",
151  n_channels, n_channels > 1 ? "s" : "");
152  if (!extra)
153  return AVERROR(ENOMEM);
154  av_dict_set(&s->metadata, "extra info", extra, AV_DICT_DONT_STRDUP_VAL);
155 
156  ADD_META_MULTIPLE_ENTRIES(instrument, ModPlug_InstrumentName);
157  ADD_META_MULTIPLE_ENTRIES(sample, ModPlug_SampleName);
158 
159  return 0;
160 }

164 static int modplug_read_header(AVFormatContext *s)
165 {
166  AVStream *st;
167  AVIOContext *pb = s->pb;
168  ModPlug_Settings settings;
169  ModPlugContext *modplug = s->priv_data;
170  int64_t sz = avio_size(pb);
171 
172  if (sz < 0) {
173  av_log(s, AV_LOG_WARNING, "Could not determine file size\n");
174  sz = modplug->max_size;
175  } else if (modplug->max_size && sz > modplug->max_size) {
176  sz = modplug->max_size;
177  av_log(s, AV_LOG_WARNING, "Max file size reach%s, allocating %"PRIi64"B "
178  "but demuxing is likely to fail due to incomplete buffer\n",
179  sz == FF_MODPLUG_DEF_FILE_SIZE ? " (see -max_size)" : "", sz);
180  }
181 
182  if (modplug->color_eval) {
183  int r = av_expr_parse(&modplug->expr, modplug->color_eval, var_names,
184  NULL, NULL, NULL, NULL, 0, s);
185  if (r < 0)
186  return r;
187  }
188 
189  modplug->buf = av_malloc(modplug->max_size);
190  if (!modplug->buf)
191  return AVERROR(ENOMEM);
192  sz = avio_read(pb, modplug->buf, sz);
193 
194  ModPlug_GetSettings(&settings);
195  settings.mChannels = 2;
196  settings.mBits = 16;
197  settings.mFrequency = 44100;
198  settings.mResamplingMode = MODPLUG_RESAMPLE_FIR; // best quality
199  settings.mLoopCount = 0; // prevents looping forever
200 
201  if (modplug->noise_reduction) settings.mFlags |= MODPLUG_ENABLE_NOISE_REDUCTION;
202  SET_OPT_IF_REQUESTED(mReverbDepth, reverb_depth, MODPLUG_ENABLE_REVERB);
203  SET_OPT_IF_REQUESTED(mReverbDelay, reverb_delay, MODPLUG_ENABLE_REVERB);
204  SET_OPT_IF_REQUESTED(mBassAmount, bass_amount, MODPLUG_ENABLE_MEGABASS);
205  SET_OPT_IF_REQUESTED(mBassRange, bass_range, MODPLUG_ENABLE_MEGABASS);
206  SET_OPT_IF_REQUESTED(mSurroundDepth, surround_depth, MODPLUG_ENABLE_SURROUND);
207  SET_OPT_IF_REQUESTED(mSurroundDelay, surround_delay, MODPLUG_ENABLE_SURROUND);
208 
209  if (modplug->reverb_depth) settings.mReverbDepth = modplug->reverb_depth;
210  if (modplug->reverb_delay) settings.mReverbDelay = modplug->reverb_delay;
211  if (modplug->bass_amount) settings.mBassAmount = modplug->bass_amount;
212  if (modplug->bass_range) settings.mBassRange = modplug->bass_range;
213  if (modplug->surround_depth) settings.mSurroundDepth = modplug->surround_depth;
214  if (modplug->surround_delay) settings.mSurroundDelay = modplug->surround_delay;
215 
216  ModPlug_SetSettings(&settings);
217 
218  modplug->f = ModPlug_Load(modplug->buf, sz);
219  if (!modplug->f) {
220  av_freep(&modplug->buf);
221  return AVERROR_INVALIDDATA;
222  }
223  st = avformat_new_stream(s, NULL);
224  if (!st)
225  return AVERROR(ENOMEM);
226  avpriv_set_pts_info(st, 64, 1, 1000);
227  st->duration = ModPlug_GetLength(modplug->f);
228  st->codecpar->codec_type = AVMEDIA_TYPE_AUDIO;
229  st->codecpar->codec_id = AV_CODEC_ID_PCM_S16LE;
230  st->codecpar->channels = settings.mChannels;
231  st->codecpar->sample_rate = settings.mFrequency;
232 
233  // timebase = 1/1000, 2ch 16bits 44.1kHz-> 2*2*44100
234  modplug->ts_per_packet = 1000*AUDIO_PKT_SIZE / (4*44100.);
235 
236  if (modplug->video_stream) {
237  AVStream *vst = avformat_new_stream(s, NULL);
238  if (!vst)
239  return AVERROR(ENOMEM);
240  avpriv_set_pts_info(vst, 64, 1, 1000);
241  vst->duration = st->duration;
242  vst->codecpar->codec_type = AVMEDIA_TYPE_VIDEO;
243  vst->codecpar->codec_id = AV_CODEC_ID_XBIN;
244  vst->codecpar->width = modplug->w << 3;
245  vst->codecpar->height = modplug->h << 3;
246  modplug->linesize = modplug->w * 3;
247  modplug->fsize = modplug->linesize * modplug->h;
248  }
249 
250  return modplug_load_metadata(s);
251 }

270 static int modplug_read_packet(AVFormatContext *s, AVPacket *pkt)
271 {
272  ModPlugContext *modplug = s->priv_data;
273  int ret;
274 
275  if (modplug->video_stream) {
276  modplug->video_switch ^= 1; // one video packet for one audio packet
277  if (modplug->video_switch) {
278  double var_values[VAR_VARS_NB];
279 
280  var_values[VAR_W ] = modplug->w;
281  var_values[VAR_H ] = modplug->h;
282  var_values[VAR_TIME ] = modplug->packet_count * modplug->ts_per_packet;
283  var_values[VAR_SPEED ] = ModPlug_GetCurrentSpeed (modplug->f);
284  var_values[VAR_TEMPO ] = ModPlug_GetCurrentTempo (modplug->f);
285  var_values[VAR_ORDER ] = ModPlug_GetCurrentOrder (modplug->f);
286  var_values[VAR_PATTERN] = ModPlug_GetCurrentPattern(modplug->f);
287  var_values[VAR_ROW ] = ModPlug_GetCurrentRow (modplug->f);
288 
289  if ((ret = av_new_packet(pkt, modplug->fsize)) < 0)
290  return ret;
291  pkt->stream_index = 1;
292  memset(pkt->data, 0, modplug->fsize);
293 
294  if (modplug->print_textinfo) {
295  char intbuf[32];
296  PRINT_INFO(0, "speed", VAR_SPEED);
297  PRINT_INFO(1, "tempo", VAR_TEMPO);
298  PRINT_INFO(2, "order", VAR_ORDER);
299  PRINT_INFO(3, "pattern", VAR_PATTERN);
300  PRINT_INFO(4, "row", VAR_ROW);
301  PRINT_INFO(5, "ts", VAR_TIME);
302  }
303 
304  if (modplug->expr) {
305  int x, y;
306  for (y = 0; y < modplug->h; y++) {
307  for (x = 0; x < modplug->w; x++) {
308  double color;
309  var_values[VAR_X] = x;
310  var_values[VAR_Y] = y;
311  color = av_expr_eval(modplug->expr, var_values, NULL);
312  pkt->data[y*modplug->linesize + x*3 + 2] |= av_clip((int)color, 0, 0xf)<<4;
313  }
314  }
315  }
316  pkt->pts = pkt->dts = var_values[VAR_TIME];
317  pkt->flags |= AV_PKT_FLAG_KEY;
318  return 0;
319  }
320  }
321 
322  if ((ret = av_new_packet(pkt, AUDIO_PKT_SIZE)) < 0)
323  return ret;
324 
325  if (modplug->video_stream)
326  pkt->pts = pkt->dts = modplug->packet_count++ * modplug->ts_per_packet;
327 
328  pkt->size = ModPlug_Read(modplug->f, pkt->data, AUDIO_PKT_SIZE);
329  if (pkt->size <= 0) {
330  return pkt->size == 0 ? AVERROR_EOF : AVERROR(EIO);
331  }
332  return 0;
333 }
