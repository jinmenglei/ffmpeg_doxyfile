54 av_cold void ff_bwdif_init_x86(BWDIFContext *bwdif)
55 {
56  YADIFContext *yadif = &bwdif->yadif;
57  int cpu_flags = av_get_cpu_flags();
58  int bit_depth = (!yadif->csp) ? 8 : yadif->csp->comp[0].depth;
59 
60  if (bit_depth <= 8) {
61 #if ARCH_X86_32
62  if (EXTERNAL_MMXEXT(cpu_flags))
63  bwdif->filter_line = ff_bwdif_filter_line_mmxext;
64 #endif /* ARCH_X86_32 */
65  if (EXTERNAL_SSE2(cpu_flags))
66  bwdif->filter_line = ff_bwdif_filter_line_sse2;
67  if (EXTERNAL_SSSE3(cpu_flags))
68  bwdif->filter_line = ff_bwdif_filter_line_ssse3;
69  } else if (bit_depth <= 12) {
70 #if ARCH_X86_32
71  if (EXTERNAL_MMXEXT(cpu_flags))
72  bwdif->filter_line = ff_bwdif_filter_line_12bit_mmxext;
73 #endif /* ARCH_X86_32 */
74  if (EXTERNAL_SSE2(cpu_flags))
75  bwdif->filter_line = ff_bwdif_filter_line_12bit_sse2;
76  if (EXTERNAL_SSSE3(cpu_flags))
77  bwdif->filter_line = ff_bwdif_filter_line_12bit_ssse3;
78  }
