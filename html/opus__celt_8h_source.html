149 static av_always_inline void celt_renormalize_vector(float *X, int N, float gain)
150 {
151  int i;
152  float g = 1e-15f;
153  for (i = 0; i < N; i++)
154  g += X[i] * X[i];
155  g = gain / sqrtf(g);
156 
157  for (i = 0; i < N; i++)
158  X[i] *= g;
159 }
