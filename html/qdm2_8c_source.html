375  int run, case_val;
376

379  if (coding_method[ch][sb][j] < 8)
380  return -1;
381  if ((coding_method[ch][sb][j] - 8) > 22) {
382  run = 1;
383  case_val = 8;
384  } else {
385  switch (switchtable[coding_method[ch][sb][j] - 8]) {
386  case 0: run = 10;
387  case_val = 10;
388  break;
389  case 1: run = 1;
390  case_val = 16;
391  break;
392  case 2: run = 5;
393  case_val = 24;
394  break;
395  case 3: run = 3;
396  case_val = 30;
397  break;
398  case 4: run = 1;
399  case_val = 30;
400  break;
401  case 5: run = 1;
402  case_val = 8;
403  break;
404  default: run = 1;
405  case_val = 8;
406  break;
407  }
408  }
409  for (k = 0; k < run; k++) {

426  }
