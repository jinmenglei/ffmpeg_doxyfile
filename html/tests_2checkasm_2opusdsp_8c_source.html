35 static void test_postfilter(int period)
36 {
37  LOCAL_ALIGNED(16, float, data0, [MAX_SIZE + 1024]);
38  LOCAL_ALIGNED(16, float, data1, [MAX_SIZE + 1024]);
39 
40  /* This filter can explode very easily, so use a tapset from the codec.
41  * In the codec these are usually multiplied by at least 0.09375f,
42  * so its outside the largest filter value, but the filter is still stable
43  * so use it. */
44  float gains[3] = { 0.3066406250f, 0.2170410156f, 0.1296386719f };
45 
46  /* The codec will always call with an offset which is aligned once
47  * (period + 2) is subtracted, but here we have to align it outselves. */
48  int offset = FFALIGN(period + 2, 4);
49 
50  declare_func(void, float *data, int period, float *gains, int len);
51 
52  randomize_float(data0, MAX_SIZE + 1024);
53  memcpy(data1, data0, (MAX_SIZE + 1024)*sizeof(float));
54 
55  call_ref(data0 + offset, period, gains, MAX_SIZE);
56  call_new(data1 + offset, period, gains, MAX_SIZE);
57 
58  if (!float_near_abs_eps_array(data0 + offset, data1 + offset, EPS, MAX_SIZE))
59  fail();
60  bench_new(data1 + offset, period, gains, MAX_SIZE);
61 }
