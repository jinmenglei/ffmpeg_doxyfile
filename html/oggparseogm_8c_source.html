172 {
173  struct ogg *ogg = s->priv_data;

175  uint8_t *p = os->buf + os->pstart;
176  int lb;
177 
178  if(*p & 8)

180 
181  lb = ((*p & 2) << 1) | ((*p >> 6) & 3);

184

186  os->psize -= lb + 1;
