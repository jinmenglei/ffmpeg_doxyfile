36 static int xvag_read_header(AVFormatContext *s)
37 {
38  unsigned offset, big_endian, codec;
39  AVStream *st;
40 
41  avio_skip(s->pb, 4);
42 
43  st = avformat_new_stream(s, NULL);
44  if (!st)
45  return AVERROR(ENOMEM);
46 
47  st->codecpar->codec_type = AVMEDIA_TYPE_AUDIO;
48 
49  offset = avio_rl32(s->pb);
50  big_endian = offset > av_bswap32(offset);
51  if (big_endian) {
52  offset = av_bswap32(offset);
53  avio_skip(s->pb, 28);
54  codec = avio_rb32(s->pb);
55  st->codecpar->channels = avio_rb32(s->pb);
56  avio_skip(s->pb, 4);
57  st->duration = avio_rb32(s->pb);
58  avio_skip(s->pb, 8);
59  st->codecpar->sample_rate = avio_rb32(s->pb);
60  } else {
61  avio_skip(s->pb, 28);
62  codec = avio_rl32(s->pb);
63  st->codecpar->channels = avio_rl32(s->pb);
64  avio_skip(s->pb, 4);
65  st->duration = avio_rl32(s->pb);
66  avio_skip(s->pb, 8);
67  st->codecpar->sample_rate = avio_rl32(s->pb);
68  }
69 
70  if (st->codecpar->sample_rate <= 0)
71  return AVERROR_INVALIDDATA;
72  if (st->codecpar->channels <= 0 || st->codecpar->channels > FF_SANE_NB_CHANNELS)
73  return AVERROR_INVALIDDATA;
74 
75  switch (codec) {
76  case 0x1c:
77  st->codecpar->codec_id = AV_CODEC_ID_ADPCM_PSX;
78  st->codecpar->block_align = 16 * st->codecpar->channels;
79  break;
80  default:
81  avpriv_request_sample(s, "codec %X", codec);
82  return AVERROR_PATCHWELCOME;
83  };
84 
85  avio_skip(s->pb, offset - avio_tell(s->pb));
86 
87  if (avio_rb16(s->pb) == 0xFFFB) {
88  st->codecpar->codec_id = AV_CODEC_ID_MP3;
89  st->codecpar->block_align = 0x1000;
90  st->need_parsing = AVSTREAM_PARSE_FULL_RAW;
91  }
92 
93  avio_skip(s->pb, -2);
94  avpriv_set_pts_info(st, 64, 1, st->codecpar->sample_rate);
95 
96  return 0;
97 }
