516  uintptr_t i = (uintptr_t)*opaque;

525 const AVFilter *avfilter_get_by_name(const char *name)
526 {
527  const AVFilter *f = NULL;
528  void *opaque = 0;
529 
530  if (!name)
531  return NULL;
532 
533  while ((f = av_filter_iterate(&opaque)))
534  if (!strcmp(f->name, name))
535  return f;
536 
537  return NULL;
538 }
