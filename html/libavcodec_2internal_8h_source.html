276 static av_always_inline float ff_exp2fi(int x) {
277  /* Normal range */
278  if (-126 <= x && x <= 128)
279  return av_int2float((x+127) << 23);
280  /* Too large */
281  else if (x > 128)
282  return INFINITY;
283  /* Subnormal numbers */
284  else if (x > -150)
285  return av_int2float(1 << (x+149));
286  /* Negligibly small */
287  else
288  return 0;
289 }
