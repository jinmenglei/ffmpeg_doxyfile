59  break;

118 
119  switch (codec_id) {

121  case AV_CODEC_ID_RV40:
122  break;
123  case AV_CODEC_ID_VP7:
124  case AV_CODEC_ID_VP8:
125  h->pred16x16[7] = ff_vp8_pred16x16_127_dc_8_msa;

127  break;
128  default:
129  h->pred16x16[PLANE_PRED8x8] =
