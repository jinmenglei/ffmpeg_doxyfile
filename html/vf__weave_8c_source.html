99 static int filter_frame(AVFilterLink *inlink, AVFrame *in)
100 {
101  AVFilterContext *ctx = inlink->dst;
102  WeaveContext *s = ctx->priv;
103  AVFilterLink *outlink = ctx->outputs[0];
104  AVFrame *out;
105  int i;
106  int weave;
107  int field1, field2;
108 
109  if (!s->prev) {
110  s->prev = in;
111  return 0;
112  }
113 
114  out = ff_get_video_buffer(outlink, outlink->w, outlink->h);
115  if (!out) {
116  av_frame_free(&in);
117  av_frame_free(&s->prev);
118  return AVERROR(ENOMEM);
119  }
120  av_frame_copy_props(out, in);
121 
122  weave = (s->double_weave && !(inlink->frame_count_out & 1));
123  field1 = weave ? s->first_field : (!s->first_field);
124  field2 = weave ? (!s->first_field) : s->first_field;
125  for (i = 0; i < s->nb_planes; i++) {
126  av_image_copy_plane(out->data[i] + out->linesize[i] * field1,
127  out->linesize[i] * 2,
128  in->data[i], in->linesize[i],
129  s->linesize[i], s->planeheight[i]);
130  av_image_copy_plane(out->data[i] + out->linesize[i] * field2,
131  out->linesize[i] * 2,
132  s->prev->data[i], s->prev->linesize[i],
133  s->linesize[i], s->planeheight[i]);
134  }
135 
136  out->pts = s->double_weave ? s->prev->pts : in->pts / 2;
137  out->interlaced_frame = 1;
138  out->top_field_first = !s->first_field;
139 
140  if (!s->double_weave)
141  av_frame_free(&in);
142  av_frame_free(&s->prev);
143  if (s->double_weave)
144  s->prev = in;
145  return ff_filter_frame(outlink, out);
146 }
