47 static int vp5_adjust(int v, int t)
48 {
49  int s2, s1 = v >> 31;
50  v ^= s1;
51  v -= s1;
52  v *= v < 2*t;
53  v -= t;
54  s2 = v >> 31;

56  v -= s2;
57  v = t - v;
58  v += s1;
59  v ^= s1;
60  return v;
