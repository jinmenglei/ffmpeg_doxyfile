51 static inline void smv_img_pnt(uint8_t *dst_data[4], uint8_t *src_data[4],
52  const int src_linesizes[4],
53  enum AVPixelFormat pix_fmt, int width, int height,
54  int nlines)
55 {
56  const AVPixFmtDescriptor *desc = av_pix_fmt_desc_get(pix_fmt);
57  int i, planes_nb = 0;
58 
59  if (desc->flags & AV_PIX_FMT_FLAG_HWACCEL)
60  return;
61 
62  for (i = 0; i < desc->nb_components; i++)
63  planes_nb = FFMAX(planes_nb, desc->comp[i].plane + 1);
64 
65  for (i = 0; i < planes_nb; i++) {
66  int h = height;
67  if (i == 1 || i == 2) {
68  h = AV_CEIL_RSHIFT(height, desc->log2_chroma_h);
69  }
70  smv_img_pnt_plane(&dst_data[i], src_data[i],
71  src_linesizes[i], h, nlines);
72  }
73  if (desc->flags & AV_PIX_FMT_FLAG_PAL ||
74  desc->flags & FF_PSEUDOPAL)
75  dst_data[1] = src_data[1];
76 }

90 static av_cold int smvjpeg_decode_init(AVCodecContext *avctx)
91 {
92  SMVJpegDecodeContext *s = avctx->priv_data;
93  AVCodec *codec;
94  AVDictionary *thread_opt = NULL;
95  int ret = 0, r;
96 
97  s->frames_per_jpeg = 0;
98 
99  s->picture[0] = av_frame_alloc();
100  if (!s->picture[0])
101  return AVERROR(ENOMEM);
102 
103  s->picture[1] = av_frame_alloc();
104  if (!s->picture[1]) {
105  av_frame_free(&s->picture[0]);
106  return AVERROR(ENOMEM);
107  }
108 
109  s->jpg.picture_ptr = s->picture[0];
110 
111  if (avctx->extradata_size >= 4)
112  s->frames_per_jpeg = AV_RL32(avctx->extradata);
113 
114  if (s->frames_per_jpeg <= 0) {
115  av_log(avctx, AV_LOG_ERROR, "Invalid number of frames per jpeg.\n");
116  ret = AVERROR_INVALIDDATA;
117  }
118 
119  codec = avcodec_find_decoder(AV_CODEC_ID_MJPEG);
120  if (!codec) {
121  av_log(avctx, AV_LOG_ERROR, "MJPEG codec not found\n");
122  smvjpeg_decode_end(avctx);
123  return AVERROR_DECODER_NOT_FOUND;
124  }
125 
126  s->avctx = avcodec_alloc_context3(codec);
127 
128  av_dict_set(&thread_opt, "threads", "1", 0);
129  s->avctx->refcounted_frames = 1;
130  s->avctx->flags = avctx->flags;
131  s->avctx->idct_algo = avctx->idct_algo;
132  if ((r = ff_codec_open2_recursive(s->avctx, codec, &thread_opt)) < 0) {
133  av_log(avctx, AV_LOG_ERROR, "MJPEG codec failed to open\n");
134  ret = r;
135  }
136  av_dict_free(&thread_opt);
137 
138  if (ret < 0)
139  smvjpeg_decode_end(avctx);
140  return ret;
141 }

143 static int smvjpeg_decode_frame(AVCodecContext *avctx, void *data, int *data_size,
144  AVPacket *avpkt)
145 {
146  const AVPixFmtDescriptor *desc;
147  SMVJpegDecodeContext *s = avctx->priv_data;
148  AVFrame* mjpeg_data = s->picture[0];
149  int i, cur_frame = 0, ret = 0;
150 
151  cur_frame = avpkt->pts % s->frames_per_jpeg;
152 
153  /* cur_frame is later used to calculate the buffer offset, so it mustn't be negative */
154  if (cur_frame < 0)
155  cur_frame += s->frames_per_jpeg;
156 
157  /* Are we at the start of a block? */
158  if (!cur_frame) {
159  av_frame_unref(mjpeg_data);
160  ret = avcodec_decode_video2(s->avctx, mjpeg_data, &s->mjpeg_data_size, avpkt);
161  if (ret < 0) {
162  s->mjpeg_data_size = 0;
163  return ret;
164  }
165  } else if (!s->mjpeg_data_size)
166  return AVERROR(EINVAL);
167 
168  desc = av_pix_fmt_desc_get(s->avctx->pix_fmt);
169  av_assert0(desc);
170 
171  if (mjpeg_data->height % (s->frames_per_jpeg << desc->log2_chroma_h)) {
172  av_log(avctx, AV_LOG_ERROR, "Invalid height\n");
173  return AVERROR_INVALIDDATA;
174  }
175 
176  /*use the last lot... */
177  *data_size = s->mjpeg_data_size;
178 
179  avctx->pix_fmt = s->avctx->pix_fmt;
180 
181  /* We shouldn't get here if frames_per_jpeg <= 0 because this was rejected
182  in init */
183  ret = ff_set_dimensions(avctx, mjpeg_data->width, mjpeg_data->height / s->frames_per_jpeg);
184  if (ret < 0) {
185  av_log(s, AV_LOG_ERROR, "Failed to set dimensions\n");
186  return ret;
187  }
188 
189  if (*data_size) {
190  s->picture[1]->extended_data = NULL;
191  s->picture[1]->width = avctx->width;
192  s->picture[1]->height = avctx->height;
193  s->picture[1]->format = avctx->pix_fmt;
194  smv_img_pnt(s->picture[1]->data, mjpeg_data->data, mjpeg_data->linesize,
195  avctx->pix_fmt, avctx->width, avctx->height, cur_frame);
196  for (i = 0; i < AV_NUM_DATA_POINTERS; i++)
197  s->picture[1]->linesize[i] = mjpeg_data->linesize[i];
198 
199  ret = av_frame_ref(data, s->picture[1]);
200  if (ret < 0)
201  return ret;
202  }
203 
204  return avpkt->size;
205 }
