27 int avpriv_codec2_mode_bit_rate(void *logctx, int mode)
28 {
29  int frame_size = avpriv_codec2_mode_frame_size(logctx, mode);
30  int block_align = avpriv_codec2_mode_block_align(logctx, mode);
31 
32  if (frame_size <= 0 || block_align <= 0) {
33  return 0;
34  }
35

37 }
