49 // refer to: http://id3.org/ID3v1
50 static int id3v1_create_tag(AVFormatContext *s, uint8_t *buf)
51 {
52  AVDictionaryEntry *tag;
53  int i, count = 0;
54 
55  memset(buf, 0, ID3v1_TAG_SIZE); /* fail safe */
56  buf[0] = 'T';
57  buf[1] = 'A';
58  buf[2] = 'G';
59  /* we knowingly overspecify each tag length by one byte to compensate for the mandatory null byte added by av_strlcpy */
60  count += id3v1_set_string(s, "TIT2", buf + 3, 30 + 1); //title
61  count += id3v1_set_string(s, "TPE1", buf + 33, 30 + 1); //author|artist
62  count += id3v1_set_string(s, "TALB", buf + 63, 30 + 1); //album
63  if ((tag = av_dict_get(s->metadata, "TYER", NULL, 0))) { //year
64  av_strlcpy(buf + 93, tag->value, 4 + 1);
65  count++;
66  } else if ((tag = av_dict_get(s->metadata, "TDRC", NULL, 0))) {
67  av_strlcpy(buf + 93, tag->value, 4 + 1);
68  count++;
69  } else if ((tag = av_dict_get(s->metadata, "TDAT", NULL, 0))) {
70  av_strlcpy(buf + 93, tag->value, 4 + 1);
71  count++;
72  }
73 
74  count += id3v1_set_string(s, "comment", buf + 97, 30 + 1);
75  if ((tag = av_dict_get(s->metadata, "TRCK", NULL, 0))) { //track
76  buf[125] = 0;
77  buf[126] = atoi(tag->value);
78  count++;
79  }
80  buf[127] = 0xFF; /* default to unknown genre */
81  if ((tag = av_dict_get(s->metadata, "TCON", NULL, 0))) { //genre
82  for(i = 0; i <= ID3v1_GENRE_MAX; i++) {
83  if (!av_strcasecmp(tag->value, ff_id3v1_genre_str[i])) {
84  buf[127] = i;
85  count++;
86  break;
87  }
88  }
89  }
90  return count;
91 }

140 /*
141  * Write an empty XING header and initialize respective data.
142  */
143 static int mp3_write_xing(AVFormatContext *s)
144 {
145  MP3Context *mp3 = s->priv_data;
146  AVCodecParameters *par = s->streams[mp3->audio_stream_idx]->codecpar;
147  AVDictionaryEntry *enc = av_dict_get(s->streams[mp3->audio_stream_idx]->metadata, "encoder", NULL, 0);
148  AVIOContext *dyn_ctx;
149  int32_t header;
150  MPADecodeHeader mpah;
151  int srate_idx, i, channels;
152  int bitrate_idx;
153  int best_bitrate_idx = -1;
154  int best_bitrate_error = INT_MAX;
155  int ret;
156  int ver = 0;
157  int bytes_needed;
158 
159  if (!(s->pb->seekable & AVIO_SEEKABLE_NORMAL) || !mp3->write_xing)
160  return 0;
161 
162  for (i = 0; i < FF_ARRAY_ELEMS(avpriv_mpa_freq_tab); i++) {
163  const uint16_t base_freq = avpriv_mpa_freq_tab[i];
164 
165  if (par->sample_rate == base_freq) ver = 0x3; // MPEG 1
166  else if (par->sample_rate == base_freq / 2) ver = 0x2; // MPEG 2
167  else if (par->sample_rate == base_freq / 4) ver = 0x0; // MPEG 2.5
168  else continue;
169 
170  srate_idx = i;
171  break;
172  }
173  if (i == FF_ARRAY_ELEMS(avpriv_mpa_freq_tab)) {
174  av_log(s, AV_LOG_WARNING, "Unsupported sample rate, not writing Xing header.\n");
175  return -1;
176  }
177 
178  switch (par->channels) {
179  case 1: channels = MPA_MONO; break;
180  case 2: channels = MPA_STEREO; break;
181  default: av_log(s, AV_LOG_WARNING, "Unsupported number of channels, "
182  "not writing Xing header.\n");
183  return -1;
184  }
185 
186  /* dummy MPEG audio header */
187  header = 0xffU << 24; // sync
188  header |= (0x7 << 5 | ver << 3 | 0x1 << 1 | 0x1) << 16; // sync/audio-version/layer 3/no crc*/
189  header |= (srate_idx << 2) << 8;
190  header |= channels << 6;
191 
192  for (bitrate_idx = 1; bitrate_idx < 15; bitrate_idx++) {
193  int bit_rate = 1000 * avpriv_mpa_bitrate_tab[ver != 3][3 - 1][bitrate_idx];
194  int error = FFABS(bit_rate - par->bit_rate);
195 
196  if (error < best_bitrate_error) {
197  best_bitrate_error = error;
198  best_bitrate_idx = bitrate_idx;
199  }
200  }
201  av_assert0(best_bitrate_idx >= 0);
202 
203  for (bitrate_idx = best_bitrate_idx; ; bitrate_idx++) {
204  int32_t mask = bitrate_idx << (4 + 8);
205  if (15 == bitrate_idx)
206  return -1;
207  header |= mask;
208 
209  ret = avpriv_mpegaudio_decode_header(&mpah, header);
210  av_assert0(ret >= 0);
211  mp3->xing_offset = xing_offtbl[mpah.lsf == 1][mpah.nb_channels == 1] + 4;
212  bytes_needed = mp3->xing_offset + XING_SIZE;
213 
214  if (bytes_needed <= mpah.frame_size)
215  break;
216 
217  header &= ~mask;
218  }
219 
220  ret = avio_open_dyn_buf(&dyn_ctx);
221  if (ret < 0)
222  return ret;
223 
224  avio_wb32(dyn_ctx, header);
225 
226  ffio_fill(dyn_ctx, 0, mp3->xing_offset - 4);
227  ffio_wfourcc(dyn_ctx, "Xing");
228  avio_wb32(dyn_ctx, 0x01 | 0x02 | 0x04 | 0x08); // frames / size / TOC / vbr scale
229 
230  mp3->size = mpah.frame_size;
231  mp3->want=1;
232  mp3->seen=0;
233  mp3->pos=0;
234 
235  avio_wb32(dyn_ctx, 0); // frames
236  avio_wb32(dyn_ctx, 0); // size
237 
238  // TOC
239  for (i = 0; i < XING_TOC_SIZE; i++)
240  avio_w8(dyn_ctx, (uint8_t)(255 * i / XING_TOC_SIZE));
241 
242  // vbr quality
243  // we write it, because some (broken) tools always expect it to be present
244  avio_wb32(dyn_ctx, 0);
245 
246  // encoder short version string
247  if (enc) {
248  uint8_t encoder_str[9] = { 0 };
249  if ( strlen(enc->value) > sizeof(encoder_str)
250  && !strcmp("Lavc libmp3lame", enc->value)) {
251  memcpy(encoder_str, "Lavf lame", 9);
252  } else
253  memcpy(encoder_str, enc->value, FFMIN(strlen(enc->value), sizeof(encoder_str)));
254 
255  avio_write(dyn_ctx, encoder_str, sizeof(encoder_str));
256  } else
257  avio_write(dyn_ctx, "Lavf\0\0\0\0\0", 9);
258 
259  avio_w8(dyn_ctx, 0); // tag revision 0 / unknown vbr method
260  avio_w8(dyn_ctx, 0); // unknown lowpass filter value
261  ffio_fill(dyn_ctx, 0, 8); // empty replaygain fields
262  avio_w8(dyn_ctx, 0); // unknown encoding flags
263  avio_w8(dyn_ctx, 0); // unknown abr/minimal bitrate
264  avio_wb24(dyn_ctx, 0); // empty encoder delay/padding
265 
266  avio_w8(dyn_ctx, 0); // misc
267  avio_w8(dyn_ctx, 0); // mp3gain
268  avio_wb16(dyn_ctx, 0); // preset
269 
270  // audio length and CRCs (will be updated later)
271  avio_wb32(dyn_ctx, 0); // music length
272  avio_wb16(dyn_ctx, 0); // music crc
273  avio_wb16(dyn_ctx, 0); // tag crc
274 
275  ffio_fill(dyn_ctx, 0, mpah.frame_size - bytes_needed);
276 
277  mp3->xing_frame_size = avio_close_dyn_buf(dyn_ctx, &mp3->xing_frame);
278  mp3->xing_frame_offset = avio_tell(s->pb);
279  avio_write(s->pb, mp3->xing_frame, mp3->xing_frame_size);
280 
281  mp3->audio_size = mp3->xing_frame_size;
282 
283  return 0;
284 }

286 /*
287  * Add a frame to XING data.
288  * Following lame's "VbrTag.c".
289  */
290 static void mp3_xing_add_frame(MP3Context *mp3, AVPacket *pkt)
291 {
292  int i;
293 
294  mp3->frames++;
295  mp3->seen++;
296  mp3->size += pkt->size;
297 
298  if (mp3->want == mp3->seen) {
299  mp3->bag[mp3->pos] = mp3->size;
300 
301  if (XING_NUM_BAGS == ++mp3->pos) {
302  /* shrink table to half size by throwing away each second bag. */
303  for (i = 1; i < XING_NUM_BAGS; i += 2)
304  mp3->bag[i >> 1] = mp3->bag[i];
305 
306  /* double wanted amount per bag. */
307  mp3->want *= 2;
308  /* adjust current position to half of table size. */
309  mp3->pos = XING_NUM_BAGS / 2;
310  }
311 
312  mp3->seen = 0;
313  }
314 }

316 static int mp3_write_audio_packet(AVFormatContext *s, AVPacket *pkt)
317 {
318  MP3Context *mp3 = s->priv_data;
319 
320  if (pkt->data && pkt->size >= 4) {
321  MPADecodeHeader mpah;
322  int ret;
323  int av_unused base;
324  uint32_t h;
325 
326  h = AV_RB32(pkt->data);
327  ret = avpriv_mpegaudio_decode_header(&mpah, h);
328  if (ret >= 0) {
329  if (!mp3->initial_bitrate)
330  mp3->initial_bitrate = mpah.bit_rate;
331  if ((mpah.bit_rate == 0) || (mp3->initial_bitrate != mpah.bit_rate))
332  mp3->has_variable_bitrate = 1;
333  } else {
334  av_log(s, AV_LOG_WARNING, "Audio packet of size %d (starting with %08"PRIX32"...) "
335  "is invalid, writing it anyway.\n", pkt->size, h);
336  }
337 
338 #ifdef FILTER_VBR_HEADERS
339  /* filter out XING and INFO headers. */
340  base = 4 + xing_offtbl[mpah.lsf == 1][mpah.nb_channels == 1];
341 
342  if (base + 4 <= pkt->size) {
343  uint32_t v = AV_RB32(pkt->data + base);
344 
345  if (MKBETAG('X','i','n','g') == v || MKBETAG('I','n','f','o') == v)
346  return 0;
347  }
348 
349  /* filter out VBRI headers. */
350  base = 4 + 32;
351 
352  if (base + 4 <= pkt->size && MKBETAG('V','B','R','I') == AV_RB32(pkt->data + base))
353  return 0;
354 #endif
355 
356  if (mp3->xing_offset) {
357  uint8_t *side_data = NULL;
358  int side_data_size;
359 
360  mp3_xing_add_frame(mp3, pkt);
361  mp3->audio_size += pkt->size;
362  mp3->audio_crc = av_crc(av_crc_get_table(AV_CRC_16_ANSI_LE),
363  mp3->audio_crc, pkt->data, pkt->size);
364 
365  side_data = av_packet_get_side_data(pkt,
366  AV_PKT_DATA_SKIP_SAMPLES,
367  &side_data_size);
368  if (side_data && side_data_size >= 10) {
369  mp3->padding = FFMAX(AV_RL32(side_data + 4) + 528 + 1, 0);
370  if (!mp3->delay)
371  mp3->delay = FFMAX(AV_RL32(side_data) - 528 - 1, 0);
372  } else {
373  mp3->padding = 0;
374  }
375  }
376  }
377 
378  return ff_raw_write_packet(s, pkt);
379 }

381 static int mp3_queue_flush(AVFormatContext *s)
382 {
383  MP3Context *mp3 = s->priv_data;
384  AVPacket pkt;
385  int ret = 0, write = 1;
386 
387  ff_id3v2_finish(&mp3->id3, s->pb, s->metadata_header_padding);
388  mp3_write_xing(s);
389 
390  while (mp3->queue) {
391  avpriv_packet_list_get(&mp3->queue, &mp3->queue_end, &pkt);
392  if (write && (ret = mp3_write_audio_packet(s, &pkt)) < 0)
393  write = 0;
394  av_packet_unref(&pkt);
395  }
396  return ret;

399 static void mp3_update_xing(AVFormatContext *s)
400 {
401  MP3Context *mp3 = s->priv_data;
402  AVReplayGain *rg;
403  uint16_t tag_crc;
404  uint8_t *toc;
405  int i, rg_size;
406  int64_t old_pos = avio_tell(s->pb);
407 
408  /* replace "Xing" identification string with "Info" for CBR files. */
409  if (!mp3->has_variable_bitrate)
410  AV_WL32(mp3->xing_frame + mp3->xing_offset, MKTAG('I', 'n', 'f', 'o'));
411 
412  AV_WB32(mp3->xing_frame + mp3->xing_offset + 8, mp3->frames);
413  AV_WB32(mp3->xing_frame + mp3->xing_offset + 12, mp3->size);
414 
415  toc = mp3->xing_frame + mp3->xing_offset + 16;
416  toc[0] = 0; // first toc entry has to be zero.
417  for (i = 1; i < XING_TOC_SIZE; ++i) {
418  int j = i * mp3->pos / XING_TOC_SIZE;
419  int seek_point = 256LL * mp3->bag[j] / mp3->size;
420  toc[i] = FFMIN(seek_point, 255);
421  }
422 
423  /* write replaygain */
424  rg = (AVReplayGain*)av_stream_get_side_data(s->streams[0], AV_PKT_DATA_REPLAYGAIN,
425  &rg_size);
426  if (rg && rg_size >= sizeof(*rg)) {
427  uint16_t val;
428 
429  AV_WB32(mp3->xing_frame + mp3->xing_offset + 131,
430  av_rescale(rg->track_peak, 1 << 23, 100000));
431 
432  if (rg->track_gain != INT32_MIN) {
433  val = FFABS(rg->track_gain / 10000) & ((1 << 9) - 1);
434  val |= (rg->track_gain < 0) << 9;
435  val |= 1 << 13;
436  AV_WB16(mp3->xing_frame + mp3->xing_offset + 135, val);
437  }
438 
439  if (rg->album_gain != INT32_MIN) {
440  val = FFABS(rg->album_gain / 10000) & ((1 << 9) - 1);
441  val |= (rg->album_gain < 0) << 9;
442  val |= 1 << 14;
443  AV_WB16(mp3->xing_frame + mp3->xing_offset + 137, val);
444  }
445  }
446 
447  /* write encoder delay/padding */
448  if (mp3->delay >= 1 << 12) {
449  mp3->delay = (1 << 12) - 1;
450  av_log(s, AV_LOG_WARNING, "Too many samples of initial padding.\n");
451  }
452  if (mp3->padding >= 1 << 12) {
453  mp3->padding = (1 << 12) - 1;
454  av_log(s, AV_LOG_WARNING, "Too many samples of trailing padding.\n");
455  }
456  AV_WB24(mp3->xing_frame + mp3->xing_offset + 141, (mp3->delay << 12) + mp3->padding);
457 
458  AV_WB32(mp3->xing_frame + mp3->xing_offset + XING_SIZE - 8, mp3->audio_size);
459  AV_WB16(mp3->xing_frame + mp3->xing_offset + XING_SIZE - 4, mp3->audio_crc);
460 
461  tag_crc = av_crc(av_crc_get_table(AV_CRC_16_ANSI_LE), 0, mp3->xing_frame, 190);
462  AV_WB16(mp3->xing_frame + mp3->xing_offset + XING_SIZE - 2, tag_crc);
463 
464  avio_seek(s->pb, mp3->xing_frame_offset, SEEK_SET);
465  avio_write(s->pb, mp3->xing_frame, mp3->xing_frame_size);
466  avio_seek(s->pb, old_pos, SEEK_SET);
467 }

469 static int mp3_write_trailer(struct AVFormatContext *s)
470 {
471  uint8_t buf[ID3v1_TAG_SIZE];
472  MP3Context *mp3 = s->priv_data;
473 
474  if (mp3->pics_to_write) {
475  av_log(s, AV_LOG_WARNING, "No packets were sent for some of the "
476  "attached pictures.\n");
477  mp3_queue_flush(s);
478  }
479 
480  /* write the id3v1 tag */
481  if (mp3->write_id3v1 && id3v1_create_tag(s, buf) > 0) {
482  avio_write(s->pb, buf, ID3v1_TAG_SIZE);
483  }
484 
485  if (mp3->xing_offset)
486  mp3_update_xing(s);
487 
488  return 0;
489 }

519 static int mp3_write_packet(AVFormatContext *s, AVPacket *pkt)
520 {
521  MP3Context *mp3 = s->priv_data;
522 
523  if (pkt->stream_index == mp3->audio_stream_idx) {
524  if (mp3->pics_to_write) {
525  /* buffer audio packets until we get all the pictures */
526  int ret = avpriv_packet_list_put(&mp3->queue, &mp3->queue_end, pkt, av_packet_ref, 0);
527 
528  if (ret < 0) {
529  av_log(s, AV_LOG_WARNING, "Not enough memory to buffer audio. Skipping picture streams\n");
530  mp3->pics_to_write = 0;
531  mp3_queue_flush(s);
532  return mp3_write_audio_packet(s, pkt);
533  }
534  } else
535  return mp3_write_audio_packet(s, pkt);
536  } else {
537  int ret;
538 
539  /* warn only once for each stream */
540  if (s->streams[pkt->stream_index]->nb_frames == 1) {
541  av_log(s, AV_LOG_WARNING, "Got more than one picture in stream %d,"
542  " ignoring.\n", pkt->stream_index);
543  }
544  if (!mp3->pics_to_write || s->streams[pkt->stream_index]->nb_frames >= 1)
545  return 0;
546 
547  if ((ret = ff_id3v2_write_apic(s, &mp3->id3, pkt)) < 0)
548  return ret;
549  mp3->pics_to_write--;
550 
551  /* flush the buffered audio packets */
552  if (!mp3->pics_to_write &&
553  (ret = mp3_queue_flush(s)) < 0)
554  return ret;
555  }
556 
557  return 0;
558 }

564 static int mp3_init(struct AVFormatContext *s)
565 {
566  MP3Context *mp3 = s->priv_data;
567  int i;
568 
569  if (mp3->id3v2_version &&
570  mp3->id3v2_version != 3 &&
571  mp3->id3v2_version != 4) {
572  av_log(s, AV_LOG_ERROR, "Invalid ID3v2 version requested: %d. Only "
573  "3, 4 or 0 (disabled) are allowed.\n", mp3->id3v2_version);
574  return AVERROR(EINVAL);
575  }
576 
577  /* check the streams -- we want exactly one audio and arbitrary number of

590  av_log(s, AV_LOG_ERROR, "Only audio streams and pictures are allowed in MP3.\n");
