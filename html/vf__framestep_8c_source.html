46 static int config_output_props(AVFilterLink *outlink)
47 {
48  AVFilterContext *ctx = outlink->src;
49  FrameStepContext *framestep = ctx->priv;
50  AVFilterLink *inlink = ctx->inputs[0];
51 
52  outlink->frame_rate =
53  av_div_q(inlink->frame_rate, (AVRational){framestep->frame_step, 1});
54 
55  av_log(ctx, AV_LOG_VERBOSE, "step:%d frame_rate:%d/%d(%f) -> frame_rate:%d/%d(%f)\n",
56  framestep->frame_step,
57  inlink->frame_rate.num, inlink->frame_rate.den, av_q2d(inlink->frame_rate),
58  outlink->frame_rate.num, outlink->frame_rate.den, av_q2d(outlink->frame_rate));
59  return 0;
