29 static av_cold int ass_decode_init(AVCodecContext *avctx)
30 {
31  avctx->subtitle_header = av_malloc(avctx->extradata_size + 1);
32  if (!avctx->subtitle_header)
33  return AVERROR(ENOMEM);
34  if (avctx->extradata_size)
35  memcpy(avctx->subtitle_header, avctx->extradata, avctx->extradata_size);
36  avctx->subtitle_header[avctx->extradata_size] = 0;
37  avctx->subtitle_header_size = avctx->extradata_size;
38  return 0;
39 }

41 static int ass_decode_frame(AVCodecContext *avctx, void *data, int *got_sub_ptr,
42  AVPacket *avpkt)
43 {
44  AVSubtitle *sub = data;
45 
46  if (avpkt->size <= 0)
47  return avpkt->size;
48 
49  sub->rects = av_malloc(sizeof(*sub->rects));
50  if (!sub->rects)
51  return AVERROR(ENOMEM);
52  sub->rects[0] = av_mallocz(sizeof(*sub->rects[0]));
53  if (!sub->rects[0])
54  return AVERROR(ENOMEM);
55  sub->num_rects = 1;
56  sub->rects[0]->type = SUBTITLE_ASS;
57  sub->rects[0]->ass = av_strdup(avpkt->data);
58  if (!sub->rects[0]->ass)
59  return AVERROR(ENOMEM);
60  *got_sub_ptr = 1;
61  return avpkt->size;
62 }
