70 static av_always_inline int vc1_unescape_buffer(const uint8_t *src, int size, uint8_t *dst)
71 {
72  int dsize = 0, i;
73 
74  if (size < 4) {
75  for (dsize = 0; dsize < size; dsize++)
76  *dst++ = *src++;
77  return size;
78  }
79  for (i = 0; i < size; i++, src++) {
80  if (src[0] == 3 && i >= 2 && !src[-1] && !src[-2] && i < size-1 && src[1] < 4) {
81  dst[dsize++] = src[1];
82  src++;
83  i++;
84  } else
85  dst[dsize++] = *src;
86  }
