52  const char *filename)
53 {
54  char buf[1024];
55  int ret;
56 
57  ret = avcodec_send_packet(dec_ctx, pkt);
58  if (ret < 0) {

60  exit(1);
61  }

63  while (ret >= 0) {
64  ret = avcodec_receive_frame(dec_ctx, frame);
65  if (ret == AVERROR(EAGAIN) || ret == AVERROR_EOF)
