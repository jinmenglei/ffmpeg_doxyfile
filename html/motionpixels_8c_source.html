95 static void mp_read_changes_map(MotionPixelsContext *mp, GetBitContext *gb, int count, int bits_len, int read_color)
96 {
97  uint16_t *pixels;
98  int offset, w, h, color = 0, x, y, i;
99 
100  while (count--) {
101  offset = get_bits_long(gb, mp->offset_bits_len);
102  w = get_bits(gb, bits_len) + 1;
103  h = get_bits(gb, bits_len) + 1;
104  if (read_color)
105  color = get_bits(gb, 15);
106  x = offset % mp->avctx->width;
107  y = offset / mp->avctx->width;
108  if (y >= mp->avctx->height)
109  continue;
110  w = FFMIN(w, mp->avctx->width - x);
111  h = FFMIN(h, mp->avctx->height - y);
112  pixels = (uint16_t *)&mp->frame->data[0][y * mp->frame->linesize[0] + x * 2];
113  while (h--) {
114  mp->changes_map[offset] = w;
115  if (read_color)
116  for (i = 0; i < w; ++i)
117  pixels[i] = color;
118  offset += mp->avctx->width;
119  pixels += mp->frame->linesize[0] / 2;
120  }
121  }
122 }

124 static int mp_get_code(MotionPixelsContext *mp, GetBitContext *gb, int size, int code)
125 {
126  while (get_bits1(gb)) {
127  ++size;
128  if (size > mp->max_codes_bits) {
129  av_log(mp->avctx, AV_LOG_ERROR, "invalid code size %d/%d\n", size, mp->max_codes_bits);
130  return AVERROR_INVALIDDATA;
131  }
132  code <<= 1;
133  if (mp_get_code(mp, gb, size, code + 1) < 0)
134  return AVERROR_INVALIDDATA;
135  }
136  if (mp->current_codes_count >= MAX_HUFF_CODES) {
137  av_log(mp->avctx, AV_LOG_ERROR, "too many codes\n");
138  return AVERROR_INVALIDDATA;
139  }
140 
141  mp->codes[mp->current_codes_count ].code = code;
142  mp->codes[mp->current_codes_count++].size = size;
143  return 0;
144 }

146 static int mp_read_codes_table(MotionPixelsContext *mp, GetBitContext *gb)
147 {
148  if (mp->codes_count == 1) {
149  mp->codes[0].delta = get_bits(gb, 4);
150  } else {
151  int i;
152  int ret;
153 
154  mp->max_codes_bits = get_bits(gb, 4);
155  for (i = 0; i < mp->codes_count; ++i)
156  mp->codes[i].delta = get_bits(gb, 4);
157  mp->current_codes_count = 0;
158  if ((ret = mp_get_code(mp, gb, 0, 0)) < 0)
159  return ret;
160  if (mp->current_codes_count < mp->codes_count) {
161  av_log(mp->avctx, AV_LOG_ERROR, "too few codes\n");
162  return AVERROR_INVALIDDATA;
163  }
164  }
165  return 0;
166 }

203 static void mp_decode_line(MotionPixelsContext *mp, GetBitContext *gb, int y)
204 {
205  YuvPixel p;
206  const int y0 = y * mp->avctx->width;
207  int w, i, x = 0;
208 
209  p = mp->vpt[y];
210  if (mp->changes_map[y0 + x] == 0) {
211  memset(mp->gradient_scale, 1, sizeof(mp->gradient_scale));
212  ++x;
213  }
214  while (x < mp->avctx->width) {
215  w = mp->changes_map[y0 + x];
216  if (w != 0) {
217  if ((y & 3) == 0) {
218  if (mp->changes_map[y0 + x + mp->avctx->width] < w ||
219  mp->changes_map[y0 + x + mp->avctx->width * 2] < w ||
220  mp->changes_map[y0 + x + mp->avctx->width * 3] < w) {
221  for (i = (x + 3) & ~3; i < x + w; i += 4) {
222  mp->hpt[((y / 4) * mp->avctx->width + i) / 4] = mp_get_yuv_from_rgb(mp, i, y);
223  }
224  }
225  }
226  x += w;
227  memset(mp->gradient_scale, 1, sizeof(mp->gradient_scale));
228  p = mp_get_yuv_from_rgb(mp, x - 1, y);
229  } else {
230  p.y += mp_gradient(mp, 0, mp_get_vlc(mp, gb));
231  p.y = av_clip_uintp2(p.y, 5);
232  if ((x & 3) == 0) {
233  if ((y & 3) == 0) {
234  p.v += mp_gradient(mp, 1, mp_get_vlc(mp, gb));
235  p.v = av_clip_intp2(p.v, 5);
236  p.u += mp_gradient(mp, 2, mp_get_vlc(mp, gb));
237  p.u = av_clip_intp2(p.u, 5);
238  mp->hpt[((y / 4) * mp->avctx->width + x) / 4] = p;
239  } else {
240  p.v = mp->hpt[((y / 4) * mp->avctx->width + x) / 4].v;
241  p.u = mp->hpt[((y / 4) * mp->avctx->width + x) / 4].u;
242  }
243  }
244  mp_set_rgb_from_yuv(mp, x, y, &p);
245  ++x;
246  }
247  }

250 static void mp_decode_frame_helper(MotionPixelsContext *mp, GetBitContext *gb)
251 {
252  YuvPixel p;
253  int y, y0;
254 
255  av_assert1(mp->changes_map[0]);
256 
257  for (y = 0; y < mp->avctx->height; ++y) {
258  if (mp->changes_map[y * mp->avctx->width] != 0) {
259  memset(mp->gradient_scale, 1, sizeof(mp->gradient_scale));
260  p = mp_get_yuv_from_rgb(mp, 0, y);
261  } else {
262  p.y += mp_gradient(mp, 0, mp_get_vlc(mp, gb));
263  p.y = av_clip_uintp2(p.y, 5);
264  if ((y & 3) == 0) {
265  p.v += mp_gradient(mp, 1, mp_get_vlc(mp, gb));
266  p.v = av_clip_intp2(p.v, 5);
267  p.u += mp_gradient(mp, 2, mp_get_vlc(mp, gb));
268  p.u = av_clip_intp2(p.u, 5);
269  }
270  mp->vpt[y] = p;
271  mp_set_rgb_from_yuv(mp, 0, y, &p);
272  }
273  }
274  for (y0 = 0; y0 < 2; ++y0)
275  for (y = y0; y < mp->avctx->height; y += 2)
276  mp_decode_line(mp, gb, y);
277 }

279 static int mp_decode_frame(AVCodecContext *avctx,
280  void *data, int *got_frame,
281  AVPacket *avpkt)
282 {
283  const uint8_t *buf = avpkt->data;
284  int buf_size = avpkt->size;
285  MotionPixelsContext *mp = avctx->priv_data;
286  GetBitContext gb;
287  int i, count1, count2, sz, ret;
288 
289  if ((ret = ff_reget_buffer(avctx, mp->frame, 0)) < 0)
290  return ret;
291 
292  /* le32 bitstream msb first */
293  av_fast_padded_malloc(&mp->bswapbuf, &mp->bswapbuf_size, buf_size);
294  if (!mp->bswapbuf)
295  return AVERROR(ENOMEM);
296  mp->bdsp.bswap_buf((uint32_t *) mp->bswapbuf, (const uint32_t *) buf,
297  buf_size / 4);
298  if (buf_size & 3)
299  memcpy(mp->bswapbuf + (buf_size & ~3), buf + (buf_size & ~3), buf_size & 3);
300  init_get_bits(&gb, mp->bswapbuf, buf_size * 8);
301 
302  memset(mp->changes_map, 0, avctx->width * avctx->height);
303  for (i = !(avctx->extradata[1] & 2); i < 2; ++i) {
304  count1 = get_bits(&gb, 12);
305  count2 = get_bits(&gb, 12);
306  mp_read_changes_map(mp, &gb, count1, 8, i);
307  mp_read_changes_map(mp, &gb, count2, 4, i);
308  }
309 
310  mp->codes_count = get_bits(&gb, 4);
311  if (mp->codes_count == 0)
312  goto end;
313 
314  if (mp->changes_map[0] == 0) {
315  *(uint16_t *)mp->frame->data[0] = get_bits(&gb, 15);
316  mp->changes_map[0] = 1;
317  }
318  if (mp_read_codes_table(mp, &gb) < 0)
319  goto end;
320 
321  sz = get_bits(&gb, 18);
322  if (avctx->extradata[0] != 5)
323  sz += get_bits(&gb, 18);
324  if (sz == 0)
325  goto end;
326 
327  if (mp->max_codes_bits <= 0)
328  goto end;
329  if (init_vlc(&mp->vlc, mp->max_codes_bits, mp->codes_count, &mp->codes[0].size, sizeof(HuffCode), 1, &mp->codes[0].code, sizeof(HuffCode), 4, 0))
330  goto end;
331  mp_decode_frame_helper(mp, &gb);
332  ff_free_vlc(&mp->vlc);
333 
334 end:
335  if ((ret = av_frame_ref(data, mp->frame)) < 0)
336  return ret;
337  *got_frame = 1;
338  return buf_size;
339 }
