53 static inline int ff_jpegls_quantize(JLSState *s, int v)
54 {
55  if (v == 0)
56  return 0;
57  if (v < 0) {
58  if (v <= -s->T3)
59  return -4;
60  if (v <= -s->T2)
61  return -3;
62  if (v <= -s->T1)
63  return -2;
64  if (v < -s->near)
65  return -1;
66  return 0;
67  } else {
68  if (v <= s->near)
69  return 0;
70  if (v < s->T1)
71  return 1;
72  if (v < s->T2)
73  return 2;
74  if (v < s->T3)
75  return 3;
76  return 4;
77  }
78 }

85 static inline void ff_jpegls_downscale_state(JLSState *state, int Q)

95 static inline int ff_jpegls_update_state_regular(JLSState *state,
96  int Q, int err)
97 {
98  if(FFABS(err) > 0xFFFF)
99  return -0x10000;
100  state->A[Q] += FFABS(err);
101  err *= state->twonear;
102  state->B[Q] += err;
103 
104  ff_jpegls_downscale_state(state, Q);
105 
106  if (state->B[Q] <= -state->N[Q]) {
107  state->B[Q] = FFMAX(state->B[Q] + state->N[Q], 1 - state->N[Q]);
108  if (state->C[Q] > -128)
109  state->C[Q]--;
110  } else if (state->B[Q] > 0) {
111  state->B[Q] = FFMIN(state->B[Q] - state->N[Q], 0);
112  if (state->C[Q] < 127)
113  state->C[Q]++;
114  }
115 
116  return err;
117 }
