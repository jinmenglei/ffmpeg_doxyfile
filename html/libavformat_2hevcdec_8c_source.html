27 static int hevc_probe(const AVProbeData *p)
28 {
29  uint32_t code = -1;
30  int vps = 0, sps = 0, pps = 0, irap = 0;
31  int i;
32 
33  for (i = 0; i < p->buf_size - 1; i++) {
34  code = (code << 8) + p->buf[i];
35  if ((code & 0xffffff00) == 0x100) {
36  uint8_t nal2 = p->buf[i + 1];
37  int type = (code & 0x7E) >> 1;
38 
39  if (code & 0x81) // forbidden and reserved zero bits
40  return 0;
41 
42  if (nal2 & 0xf8) // reserved zero
43  return 0;
44 
45  switch (type) {
46  case HEVC_NAL_VPS: vps++; break;
47  case HEVC_NAL_SPS: sps++; break;
48  case HEVC_NAL_PPS: pps++; break;
49  case HEVC_NAL_BLA_N_LP:
50  case HEVC_NAL_BLA_W_LP:
51  case HEVC_NAL_BLA_W_RADL:
52  case HEVC_NAL_CRA_NUT:
53  case HEVC_NAL_IDR_N_LP:
54  case HEVC_NAL_IDR_W_RADL: irap++; break;
55  }
56  }
57  }
58 
59  if (vps && sps && pps && irap)
60  return AVPROBE_SCORE_EXTENSION + 1; // 1 more than .mpg
61  return 0;
62 }
