60 /* fast header check for resync */
61 static inline int ff_mpa_check_header(uint32_t header){
62  /* header */
63  if ((header & 0xffe00000) != 0xffe00000)
64  return -1;
65  /* version check */
66  if ((header & (3<<19)) == 1<<19)
67  return -1;
68  /* layer check */
69  if ((header & (3<<17)) == 0)
70  return -1;
71  /* bit rate */
72  if ((header & (0xf<<12)) == 0xf<<12)
73  return -1;
74  /* frequency */
75  if ((header & (3<<10)) == 3<<10)
76  return -1;
77  return 0;
78 }
